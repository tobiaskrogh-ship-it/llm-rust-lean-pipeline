/// For a given list of input numbers, calculate Mean Absolute Deviation
/// around the mean:  MAD = average | x - x_mean |
pub fn mean_absolute_deviation(numbers: &[f64]) -> f64 {
    let n = numbers.len() as f64;
    let mut sum: f64 = 0.0;
    for &x in numbers {
        sum += x;
    }
    let mean = sum / n;
    let mut acc: f64 = 0.0;
    for &x in numbers {
        acc += (x - mean).abs();
    }
    acc / n
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    /// Failure condition: an empty slice has n = 0, so the function divides by
    /// zero and the result is NaN. Pins down the only documented edge case.
    #[test]
    fn empty_slice_returns_nan() {
        assert!(mean_absolute_deviation(&[]).is_nan());
    }

    proptest! {
        /// Postcondition (concrete value): for a single-element slice the lone
        /// element IS the mean, so MAD = 0 exactly. This is exact in floating
        /// point because `(x - x).abs() == 0.0` for any finite `x`.
        #[test]
        fn single_element_is_zero(x in -1.0e6f64..1.0e6) {
            prop_assert_eq!(mean_absolute_deviation(&[x]), 0.0);
        }

        /// Postcondition (concrete value): a constant sequence has zero spread,
        /// hence MAD = 0 (up to floating-point rounding in the mean). Catches
        /// implementations that return the mean itself, or that skip the
        /// subtraction.
        #[test]
        fn constant_sequence_is_zero(c in -1.0e3f64..1.0e3, n in 1usize..32) {
            let xs = vec![c; n];
            let result = mean_absolute_deviation(&xs);
            // Rounding when computing `sum / n` can perturb `mean` slightly
            // away from `c`, so allow a tiny absolute tolerance.
            prop_assert!(result.abs() < 1.0e-9);
        }

        /// Concrete contract verification: for two elements MAD([a, b]) reduces
        /// to |a - b| / 2 (closed form). This pins down the exact formula and
        /// catches a missing `abs`, an off-by-one in the divisor (e.g. n - 1),
        /// or returning variance / standard deviation instead of MAD.
        #[test]
        fn two_element_formula(a in -1.0e3f64..1.0e3, b in -1.0e3f64..1.0e3) {
            let got = mean_absolute_deviation(&[a, b]);
            let expected = (a - b).abs() / 2.0;
            prop_assert!((got - expected).abs() < 1.0e-9);
        }

        /// Postcondition: MAD is an average of non-negative terms `|x - mean|`,
        /// so the result must be non-negative for every non-empty finite input.
        #[test]
        fn non_negative(xs in proptest::collection::vec(-1.0e6f64..1.0e6, 1..32)) {
            prop_assert!(mean_absolute_deviation(&xs) >= 0.0);
        }
    }
}

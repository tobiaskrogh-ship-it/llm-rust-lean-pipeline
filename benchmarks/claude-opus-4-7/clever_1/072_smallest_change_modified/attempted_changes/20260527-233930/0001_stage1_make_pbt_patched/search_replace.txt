    fn naive(arr: &[i64]) -> i64 {
        let n = arr.len();
        let mut count = 0i64;
        for i in 0..n / 2 {
            if arr[i] != arr[n - 1 - i] {

            }

        count
    }

    fn is_palindrome(arr: &[i64]) -> bool {
        let n = arr.len();
        for i in 0..n / 2 {
            if arr[i] != arr[n - 1 - i] {
                return false;
            }
        }

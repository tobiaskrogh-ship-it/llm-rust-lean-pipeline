    proptest! {
        /// Postcondition — recurrence for `n ≥ 3`.
        /// Bound kept well below the i64 overflow threshold (`fibfib`
        /// grows ~1.84^n; i64 fits up to roughly n = 75).
        #[test]
        fn recurrence_on_nonneg_range(n in 3i64..=60) {
            prop_assert_eq!(fibfib(n), fibfib(n - 1) + fibfib(n - 2) + fibfib(n - 3));
        }

        /// Postcondition — negative inputs collapse to 0.
        /// This is an independent contract clause from the recurrence:
        /// a buggy implementation that panicked, recursed, or returned
        /// some other sentinel on negatives would pass the other tests.
        #[test]
        fn negative_inputs_are_zero(n in i64::MIN..0) {
            prop_assert_eq!(fibfib(n), 0);
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    #[test]
    fn known() {
        assert_eq!(digits(0), 0);
        assert_eq!(digits(1), 1);
        assert_eq!(digits(4), 0);          // all even
        assert_eq!(digits(235), 15);       // 3 * 5
        assert_eq!(digits(2468), 0);
        assert_eq!(digits(2222), 0);
    }

    /// Independent reference: iterative product of the odd decimal digits of `n`,
    /// returning 0 when `n` has no odd digit (including `n == 0`).
    ///
    /// No overflow check is needed: for any `u64` input the number of decimal
    /// digits is at most 20, so the product of odd digits is at most
    /// 9^20 = 12_157_665_459_056_928_801 < u64::MAX.
    fn reference(n: u64) -> u64 {
        if n == 0 {
            return 0;
        }
        let mut m = n;
        let mut prod: u64 = 1;
        let mut any_odd = false;
        while m > 0 {
            let d = m % 10;
            if d % 2 == 1 {
                prod *= d;
                any_odd = true;
            }
            m /= 10;
        }
        if any_odd { prod } else { 0 }
    }

    proptest! {
        /// Main postcondition: `digits(n)` equals the product of the odd decimal
        /// digits of `n`, with 0 in the all-even / zero case. Tested against an
        /// independent iterative reference for arbitrary `u64` inputs.
        #[test]
        fn prop_matches_reference(n in any::<u64>()) {
            prop_assert_eq!(digits(n), reference(n));
        }

        /// Empty-odd convention: when every decimal digit of `n` is even
        /// (including the special case `n == 0`), `digits(n)` is *exactly 0*,
        /// not the empty product 1. We generate `n` from digits drawn only from
        /// {0, 2, 4, 6, 8}. Length is capped at 18 so the composed value
        /// always fits in a `u64` (worst case 888…8 with 18 eights < u64::MAX).
        #[test]
        fn prop_all_even_digits_returns_zero(
            ds in proptest::collection::vec(
                prop_oneof![Just(0u64), Just(2u64), Just(4u64), Just(6u64), Just(8u64)],
                0..19usize,
            )
        ) {
            let n = ds.iter().fold(0u64, |acc, &d| acc * 10 + d);
            prop_assert_eq!(digits(n), 0);
        }
    }
}

pub fn prime_fib(n: u64) -> u64 {
    prime_fib_at(n, 1, 1, 0)
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    // ---- Oracles (independent of the implementation under test) ----

    /// Trial-division primality test.
    fn is_prime(n: u64) -> bool {
        if n < 2 {
            return false;
        }
        let mut d: u64 = 2;
        while d.saturating_mul(d) <= n {
            if n % d == 0 {
                return false;
            }
            d += 1;
        }
        true
    }

    /// Iterate the Fibonacci sequence seeded with (1, 1) — matching the
    /// seed used by `prime_fib_at` — and check whether `n` appears.
    fn is_fibonacci(n: u64) -> bool {
        if n == 1 {
            return true;
        }
        let (mut a, mut b): (u64, u64) = (1, 1);
        loop {
            let c = match a.checked_add(b) {
                Some(c) => c,
                None => return false,
            };
            if c == n {
                return true;
            }
            if c > n {
                return false;
            }
            a = b;
            b = c;
        }
    }

    // ---- Anchor for the "n-th" claim ----

    /// `prime_fib(0)` must be 2, the smallest prime Fibonacci. This pins the
    /// base of the enumeration; combined with `no_prime_fib_skipped` below,
    /// it fixes the indexing of the whole sequence.
    #[test]
    fn prime_fib_zero_is_two() {
        assert_eq!(prime_fib(0), 2);
    }

    proptest! {
        /// Postcondition: the result is prime.
        #[test]
        fn result_is_prime(n in 0u64..=6) {
            let v = prime_fib(n);
            prop_assert!(is_prime(v), "prime_fib({}) = {} is not prime", n, v);
        }

        /// Postcondition: the result is a Fibonacci number.
        #[test]
        fn result_is_fibonacci(n in 0u64..=6) {
            let v = prime_fib(n);
            prop_assert!(
                is_fibonacci(v),
                "prime_fib({}) = {} is not a Fibonacci number",
                n,
                v
            );
        }

        /// Postcondition: `prime_fib(n+1)` is the *next* prime Fibonacci
        /// strictly greater than `prime_fib(n)`. Equivalently: the sequence
        /// is strictly increasing AND no prime Fibonacci is skipped between
        /// consecutive indices. This pins down the "n-th" part of the
        /// contract (which is independent of "prime" and "Fibonacci").
        #[test]
        fn no_prime_fib_skipped(n in 0u64..=5) {
            let lo = prime_fib(n);
            let hi = prime_fib(n + 1);
            prop_assert!(lo < hi, "prime_fib must be strictly increasing");

            // Walk Fibonacci numbers strictly between lo and hi; none may be prime.
            let (mut a, mut b): (u64, u64) = (1, 1);
            loop {
                let c = a + b;
                if c >= hi {
                    break;
                }
                if c > lo {
                    prop_assert!(
                        !is_prime(c),
                        "Fibonacci {} lies strictly between prime_fib({}) = {} \
                         and prime_fib({}) = {} but is prime",
                        c, n, lo, n + 1, hi
                    );
                }
                a = b;
                b = c;
            }
        }
    }
}

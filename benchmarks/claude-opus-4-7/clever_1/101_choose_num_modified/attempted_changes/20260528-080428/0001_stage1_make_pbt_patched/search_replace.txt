#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    #[test]
    fn known() {
        assert_eq!(choose_num(12, 15), 14);
        assert_eq!(choose_num(13, 12), -1);
        assert_eq!(choose_num(0, 0), 0);
        assert_eq!(choose_num(1, 1), -1);
    }

    proptest! {
        /// Failure characterization (both directions): `choose_num` returns
        /// the `-1` sentinel exactly when no even integer exists in `[x, y]`.
        /// No even exists iff `x > y` (empty range) or `x == y` with `x` odd
        /// (single-odd range); whenever `x < y` the two-or-more consecutive
        /// integers contain at least one even number, so a valid answer
        /// exists.
        #[test]
        fn returns_neg1_iff_no_even_in_range(
            x in -1_000_000i64..1_000_000,
            y in -1_000_000i64..1_000_000,
        ) {
            let r = choose_num(x, y);
            let no_even_in_range = x > y || (x == y && x % 2 != 0);
            prop_assert_eq!(r == -1, no_even_in_range);
        }

        /// Postcondition (evenness): a non-sentinel result is an even integer.
        /// Catches a buggy implementation that might return `y` (or any odd
        /// candidate) in cases where the largest in-range even is `y - 1`.
        #[test]
        fn result_is_even_when_valid(
            x in -1_000_000i64..1_000_000,
            y in -1_000_000i64..1_000_000,
        ) {
            let r = choose_num(x, y);
            prop_assume!(r != -1);
            prop_assert_eq!(r % 2, 0);
        }

        /// Postcondition (range): a non-sentinel result lies inside `[x, y]`.
        /// Catches an implementation that returns an even integer outside
        /// the requested window (e.g. always 0, or `y + 1` rounded down).
        #[test]
        fn result_in_range_when_valid(
            x in -1_000_000i64..1_000_000,
            y in -1_000_000i64..1_000_000,
        ) {
            let r = choose_num(x, y);
            prop_assume!(r != -1);
            prop_assert!(x <= r && r <= y);
        }

        /// Postcondition (maximality): no larger even integer exists in
        /// `[x, y]`. Since the result is even (by the evenness test), the
        /// next-larger even candidate is `r + 2`; requiring `r + 2 > y`
        /// rules out any greater in-range even. Catches an implementation
        /// that returns *some* even in range but not the *largest* one
        /// (e.g. returns `x` when `x` is even).
        #[test]
        fn result_is_maximal_when_valid(
            x in -1_000_000i64..1_000_000,
            y in -1_000_000i64..1_000_000,
        ) {
            let r = choose_num(x, y);
            prop_assume!(r != -1);
            prop_assert!(r + 2 > y);
        }
    }
}

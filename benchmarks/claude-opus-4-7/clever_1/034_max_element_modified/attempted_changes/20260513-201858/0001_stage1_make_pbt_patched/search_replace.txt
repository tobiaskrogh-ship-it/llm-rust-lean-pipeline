pub fn max_element(l: &[i64]) -> i64 {
    if l.is_empty() {
        0
    } else {
        max_at(l, 1, l[0])
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    // Postcondition (empty case): the empty list maps to the sentinel value 0.
    // Pinned as a concrete test because it is a documented special case rather
    // than a property of arbitrary inputs.
    #[test]
    fn empty_list_returns_zero() {
        let empty: [i64; 0] = [];
        assert_eq!(max_element(&empty), 0);
    }

    proptest! {
        // Postcondition (non-empty, part 1 of "greatest element"):
        // the result is actually one of the list's elements. A buggy impl that
        // returned `i64::MAX` or a value derived but not present (e.g. max+1)
        // would still satisfy the >= bound below, so this claim is independent.
        #[test]
        fn result_is_an_element_of_the_list(l in proptest::collection::vec(any::<i64>(), 1..32)) {
            let m = max_element(&l);
            prop_assert!(l.contains(&m));
        }

        // Postcondition (non-empty, part 2 of "greatest element"):
        // the result dominates every element. A buggy impl that returned
        // `l[0]` (or any element other than the maximum) would still satisfy
        // "in the list", so this claim is independent of the one above.
        #[test]
        fn result_is_an_upper_bound(l in proptest::collection::vec(any::<i64>(), 1..32)) {
            let m = max_element(&l);
            for &x in &l {
                prop_assert!(m >= x);
            }
        }

        // Totality: the function does not panic on any `&[i64]` input
        // (including the empty slice). proptest discharges this implicitly
        // by simply running max_element to completion.
        #[test]
        fn never_panics(l in proptest::collection::vec(any::<i64>(), 0..32)) {
            let _ = max_element(&l);
        }
    }
}

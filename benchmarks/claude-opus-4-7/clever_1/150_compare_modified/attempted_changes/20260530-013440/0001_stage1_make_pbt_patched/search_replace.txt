#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    #[test]
    fn known() {
        assert_eq!(compare(&[1, 2, 3, 4, 5, 1], &[1, 2, 3, 4, 2, -2]), vec![0, 0, 0, 0, 3, 3]);
        assert_eq!(compare(&[0, 5, 0, 0, 0, 4], &[4, 1, 1, 0, 0, -2]), vec![4, 4, 1, 0, 0, 6]);
    }

    // Use a bounded range so the absolute difference never overflows i64.
    // The contract `result[i] = |scores[i] - guesses[i]|` only makes sense
    // when this absolute value is representable.
    fn pair_strategy() -> impl Strategy<Value = (Vec<i64>, Vec<i64>)> {
        (
            prop::collection::vec(i32::MIN as i64..=i32::MAX as i64, 0..32),
            prop::collection::vec(i32::MIN as i64..=i32::MAX as i64, 0..32),
        )
    }

    proptest! {
        // Postcondition (length): the output length is the min of the input lengths.
        #[test]
        fn length_is_min_of_inputs((s, g) in pair_strategy()) {
            let out = compare(&s, &g);
            prop_assert_eq!(out.len(), s.len().min(g.len()));
        }

        // Postcondition (elements): for every valid index i,
        //   out[i] = |scores[i] - guesses[i]|.
        // This is the core semantic claim of the function and pins down
        // the value at every position.
        #[test]
        fn element_is_absolute_difference((s, g) in pair_strategy()) {
            let out = compare(&s, &g);
            for i in 0..out.len() {
                let expected = (s[i] - g[i]).abs();
                prop_assert_eq!(out[i], expected);
            }
        }
    }
}

/-- `insert_asc_at` preserves `all_odd` when `x` is odd: by strong induction
    on `arr.val.size - i.toNat`. -/
private theorem insert_asc_at_preserves_all_odd
    (arr : RustSlice u64) (x : u64) (h_x : x.toNat % 2 = 1)
    (h_arr_odd : ∀ j : Nat, ∀ (hj : j < arr.val.size), (arr.val[j]'hj).toNat % 2 = 1) :
    ∀ (i : usize) (done : Bool) (acc : alloc.vec.Vec u64 alloc.alloc.Global),
      all_odd acc →
      ∀ v, clever_122_get_odd_collatz.insert_asc_at arr x i done acc = RustM.ok v →
        all_odd v := by
  intro i done acc h_acc v hres
  induction hk : (arr.val.size - i.toNat) using Nat.strongRecOn generalizing i done acc with
  | _ k ih =>
    sorry

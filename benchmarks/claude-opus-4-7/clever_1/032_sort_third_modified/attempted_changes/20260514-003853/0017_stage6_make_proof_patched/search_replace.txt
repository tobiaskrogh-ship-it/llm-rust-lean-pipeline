private theorem collect_thirds_size
    (l : RustSlice i64)
    (sorted : alloc.vec.Vec i64 alloc.alloc.Global)
    (hct : clever_032_sort_third.collect_thirds l (0 : usize)
        ⟨(List.nil : List i64).toArray, by grind⟩ = RustM.ok sorted) :
    sorted.val.size = (l.val.size + 2) / 3 := by
  -- Strong-induction-friendly generalisation. The invariant
  -- `acc.val.size = (i.toNat + 2) / 3` holds at start (acc = [], i = 0 ⇒ 0 = 0)
  -- and is preserved by both branches:
  --   * non-third step (i % 3 ≠ 0): acc unchanged, i → i+1; (i+2)/3 = (i+3)/3
  --     since i % 3 ∈ {1, 2}.
  --   * third step (i % 3 = 0): acc grows by 1 (insert_sorted adds one element),
  --     i → i+1; (i+2)/3 + 1 = (i+3)/3 since i % 3 = 0.
  suffices h_gen :
      ∀ (n : Nat) (i : usize) (acc sorted' : alloc.vec.Vec i64 alloc.alloc.Global),
        l.val.size - i.toNat ≤ n →
        i.toNat ≤ l.val.size →
        acc.val.size = (i.toNat + 2) / 3 →
        clever_032_sort_third.collect_thirds l i acc = RustM.ok sorted' →
        sorted'.val.size = (l.val.size + 2) / 3 by
    let acc0 : alloc.vec.Vec i64 alloc.alloc.Global :=
      ⟨(List.nil : List i64).toArray, by grind⟩
    have h_acc0_size : acc0.val.size = ((0 : usize).toNat + 2) / 3 := by
      show (0 : Nat) = 2 / 3
      decide
    have h_meas : l.val.size - (0 : usize).toNat ≤ l.val.size := by
      show l.val.size - 0 ≤ l.val.size; omega
    have h_i_le : (0 : usize).toNat ≤ l.val.size := by
      show 0 ≤ l.val.size; omega
    exact h_gen l.val.size (0 : usize) acc0 sorted h_meas h_i_le h_acc0_size hct
  /- Stuck at the body of `h_gen` — needs:
       (1) `collect_thirds` step lemmas (OOB / third / non-third) to unfold
           one iteration of the tail recursion;
       (2) `insert_sorted_size_correct : insert_sorted v x = ok v' →
              v'.val.size = v.val.size + 1` to discharge the third-step
           invariant preservation.
     Neither is developed in this file; (2) is the genuinely new piece. -/
  sorry

    (l : RustSlice i64)
    (sorted : alloc.vec.Vec i64 alloc.alloc.Global)
    (hct : clever_032_sort_third.collect_thirds l (0 : usize)

    ∀ x : i64,
      (sorted.val.toList.count x) = ((third_values l.val).count x) := by
  intro x
  -- Reduce to a strong-induction-friendly statement parameterised over
  -- (i, acc), tracking the "delta count" invariant
  --   sorted'.val.toList.count x = acc.val.toList.count x +
  --     ((third_values l.val).count x - (third_values_aux l.val i.toNat).count x).

  suffices h_gen :
      ∀ (i : usize) (acc sorted' : alloc.vec.Vec i64 alloc.alloc.Global),
        i.toNat ≤ l.val.size →
        clever_032_sort_third.collect_thirds l i acc = RustM.ok sorted' →
        (sorted'.val.toList.count x : Int) =
          (acc.val.toList.count x : Int) +
            ((third_values l.val).count x : Int) -
              ((third_values_aux l.val i.toNat).count x : Int) by

      ⟨(List.nil : List i64).toArray, by grind⟩

/-! ## Strong induction for the third-position invariant of `rebuild_at`.

Captures: when `j.toNat = (i.toNat + 2) / 3` (the count of thirds processed
so far) and the accumulator's third-aligned entries already match `sorted`'s
prefix, the final `v` continues to satisfy `v.val[k] = sorted.val[k/3]` at
all third-aligned indices. This is the missing strengthening that closes
`rebuild_at_third_value_stub` directly.

Note: `(i.toNat + 2) / 3 = ⌈i.toNat / 3⌉`. At `i.toNat = 0` this is `0`; at
each third-step (`i % 3 = 0`) it grows by `1`; at each non-third-step it
stays the same. The Nat-division omega extension closes both arithmetic
identities directly. -/

private theorem rebuild_at_correct_third (l sorted : RustSlice i64) :
    ∀ (n : Nat) (i j : usize) (acc : alloc.vec.Vec i64 alloc.alloc.Global)
      (v : alloc.vec.Vec i64 alloc.alloc.Global),
      l.val.size - i.toNat ≤ n →
      i.toNat ≤ l.val.size →
      acc.val.size = i.toNat →
      j.toNat = (i.toNat + 2) / 3 →
      (∀ (k : Nat) (hk : k < acc.val.size),
          k % 3 = 0 →
          ∃ (hj : (k / 3) < sorted.val.size),
              acc.val[k]'hk = sorted.val[k / 3]'hj) →
      clever_032_sort_third.rebuild_at l sorted i j acc = RustM.ok v →
      ∀ (k : Nat) (hk_v : k < v.val.size),
          k % 3 = 0 →
          ∃ (hj : (k / 3) < sorted.val.size),
              v.val[k]'hk_v = sorted.val[k / 3]'hj := by
  intro n
  induction n with
  | zero =>
    intro i j acc v hm hi_le h_acc_size h_j_eq h_acc_inv hres
    have hi_ge : l.val.size ≤ i.toNat := by omega
    rw [rebuild_at_oob l sorted i j acc hi_ge] at hres
    injection hres with h_eq
    injection h_eq with h_eq'
    subst h_eq'
    intro k hk_v hkmod
    exact h_acc_inv k hk_v hkmod
  | succ n ih =>
    intro i j acc v hm hi_le h_acc_size h_j_eq h_acc_inv hres
    by_cases hi_ge : l.val.size ≤ i.toNat
    · rw [rebuild_at_oob l sorted i j acc hi_ge] at hres
      injection hres with h_eq
      injection h_eq with h_eq'
      subst h_eq'
      intro k hk_v hkmod
      exact h_acc_inv k hk_v hkmod
    · have hi_lt : i.toNat < l.val.size := Nat.lt_of_not_le hi_ge
      have h_size_lt : l.val.size < USize64.size := l.size_lt_usizeSize
      have h_usize_size : USize64.size = 2 ^ 64 := usize_size_eq
      have h_no_ov_i : i.toNat + 1 < 2^64 := by
        rw [h_usize_size] at h_size_lt; omega
      have h_i1 : (i + 1).toNat = i.toNat + 1 := usize_add_one_toNat i h_no_ov_i
      have h_acc_succ : acc.val.size + 1 < USize64.size := by
        rw [h_acc_size, h_usize_size]; omega
      have h_i1_le : (i + 1).toNat ≤ l.val.size := by rw [h_i1]; omega
      have h_meas : l.val.size - (i + 1).toNat ≤ n := by rw [h_i1]; omega
      by_cases hmod_i : i.toNat % 3 = 0
      · -- Third branch.
        by_cases hj_lt : j.toNat < sorted.val.size
        · rw [rebuild_at_step_third l sorted i j acc hi_lt hmod_i hj_lt h_acc_succ] at hres
          -- j.toNat = i.toNat / 3 since i.toNat % 3 = 0 and j.toNat = (i+2)/3.
          have h_j_val : j.toNat = i.toNat / 3 := by rw [h_j_eq]; omega
          have h_sorted_size_lt : sorted.val.size < USize64.size := sorted.size_lt_usizeSize
          have h_no_ov_j : j.toNat + 1 < 2 ^ 64 := by
            rw [h_usize_size] at h_sorted_size_lt; omega
          have h_j1 : (j + 1).toNat = j.toNat + 1 := usize_add_one_toNat j h_no_ov_j
          have h_j1_eq : (j + 1).toNat = ((i + 1).toNat + 2) / 3 := by
            rw [h_j1, h_i1, h_j_eq]; omega
          have h_acc'_size :
              (push_one acc (sorted.val[j.toNat]'hj_lt) h_acc_succ).val.size = (i + 1).toNat := by
            show (acc.val ++ #[_]).size = (i + 1).toNat
            rw [Array.size_append, h_i1, h_acc_size]; rfl
          have h_acc'_inv :
              ∀ (k : Nat)
                (hk : k < (push_one acc (sorted.val[j.toNat]'hj_lt) h_acc_succ).val.size),
                k % 3 = 0 →
                ∃ (hj' : (k / 3) < sorted.val.size),
                  (push_one acc (sorted.val[j.toNat]'hj_lt) h_acc_succ).val[k]'hk =
                    sorted.val[k / 3]'hj' := by
            intro k hk hkmod
            show ∃ (hj' : (k / 3) < sorted.val.size),
                  ((acc.val ++ #[sorted.val[j.toNat]'hj_lt])[k]'hk) = sorted.val[k / 3]'hj'
            by_cases hk_lt : k < acc.val.size
            · obtain ⟨hjk, hacc_eq⟩ := h_acc_inv k hk_lt hkmod
              refine ⟨hjk, ?_⟩
              rw [Array.getElem_append_left hk_lt]
              exact hacc_eq
            · -- k = acc.val.size = i.toNat.
              have h_size_raw :
                  (acc.val ++ #[sorted.val[j.toNat]'hj_lt]).size = acc.val.size + 1 := by
                rw [Array.size_append]; rfl
              have hk_eq : k = acc.val.size := by
                have : k < acc.val.size + 1 := by rw [← h_size_raw]; exact hk
                omega
              -- k/3 = i.toNat/3 = j.toNat.
              have h_k_div : k / 3 = j.toNat := by rw [hk_eq, h_acc_size, ← h_j_val]
              refine ⟨?_, ?_⟩
              · rw [h_k_div]; exact hj_lt
              · subst hk_eq
                rw [Array.getElem_append_right (Nat.le_refl _)]
                simp only [Nat.sub_self]
                show sorted.val[j.toNat]'hj_lt = sorted.val[acc.val.size / 3]'_
                have h_acc_div : acc.val.size / 3 = j.toNat := by
                  rw [h_acc_size, ← h_j_val]
                exact getElem_congr_idx h_acc_div.symm
          exact ih (i + 1) (j + 1) _ v h_meas h_i1_le h_acc'_size h_j1_eq h_acc'_inv hres
        · -- j out of range: contradicts hres = ok.
          exfalso
          have hj_ge : sorted.val.size ≤ j.toNat := Nat.le_of_not_lt hj_lt
          rw [rebuild_at_third_idx_fail l sorted i j acc hi_lt hmod_i hj_ge] at hres
          cases hres
      · -- Non-third branch. j unchanged.
        rw [rebuild_at_step_nonthird l sorted i j acc hi_lt hmod_i h_acc_succ] at hres
        have h_j1_eq : j.toNat = ((i + 1).toNat + 2) / 3 := by
          rw [h_i1, h_j_eq]; omega
        have h_acc'_size :
            (push_one acc (l.val[i.toNat]'hi_lt) h_acc_succ).val.size = (i + 1).toNat := by
          show (acc.val ++ #[_]).size = (i + 1).toNat
          rw [Array.size_append, h_i1, h_acc_size]; rfl
        have h_acc'_inv :
            ∀ (k : Nat)
              (hk : k < (push_one acc (l.val[i.toNat]'hi_lt) h_acc_succ).val.size),
              k % 3 = 0 →
              ∃ (hj' : (k / 3) < sorted.val.size),
                (push_one acc (l.val[i.toNat]'hi_lt) h_acc_succ).val[k]'hk =
                  sorted.val[k / 3]'hj' := by
          intro k hk hkmod
          show ∃ (hj' : (k / 3) < sorted.val.size),
                ((acc.val ++ #[l.val[i.toNat]'hi_lt])[k]'hk) = sorted.val[k / 3]'hj'
          by_cases hk_lt : k < acc.val.size
          · obtain ⟨hjk, hacc_eq⟩ := h_acc_inv k hk_lt hkmod
            refine ⟨hjk, ?_⟩
            rw [Array.getElem_append_left hk_lt]
            exact hacc_eq
          · -- k = acc.val.size = i.toNat, but i.toNat % 3 ≠ 0 contradicts hkmod.
            exfalso
            have h_size_raw :
                (acc.val ++ #[l.val[i.toNat]'hi_lt]).size = acc.val.size + 1 := by
              rw [Array.size_append]; rfl
            have hk_eq : k = acc.val.size := by
              have : k < acc.val.size + 1 := by rw [← h_size_raw]; exact hk
              omega
            rw [hk_eq, h_acc_size] at hkmod
            exact hmod_i hkmod
        exact ih (i + 1) j _ v h_meas h_i1_le h_acc'_size h_j1_eq h_acc'_inv hres

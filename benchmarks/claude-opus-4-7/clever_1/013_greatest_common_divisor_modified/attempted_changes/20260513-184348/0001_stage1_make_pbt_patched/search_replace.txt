/// Return the greatest common divisor of two non-negative integers.
pub fn greatest_common_divisor(a: u64, b: u64) -> u64 {
    let mut a = a;
    let mut b = b;
    while b != 0 {
        hax_lib::loop_decreases!(b);
        let t = b;
        b = a % b;
        a = t;
    }
    a
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    proptest! {
        // Postcondition (common divisor): the result divides both inputs.
        // By convention gcd(0, 0) = 0, which is the only case where the
        // result is 0.
        #[test]
        fn result_is_a_common_divisor(a: u64, b: u64) {
            let g = greatest_common_divisor(a, b);
            if g == 0 {
                prop_assert_eq!(a, 0);
                prop_assert_eq!(b, 0);
            } else {
                prop_assert_eq!(a % g, 0);
                prop_assert_eq!(b % g, 0);
            }
        }

        // Postcondition (greatest): every common divisor of a and b must
        // divide the result. We witness this by constructing inputs that
        // share a known divisor d, then checking gcd(a, b) is a multiple
        // of d. A buggy implementation that returned, say, the smaller
        // input or 1 would fail this.
        #[test]
        fn result_is_divisible_by_every_common_divisor(
            d in 1u64..=1_000_000,
            m in 0u64..=1_000_000,
            n in 0u64..=1_000_000,
        ) {
            let a = d * m;
            let b = d * n;
            let g = greatest_common_divisor(a, b);
            if g != 0 {
                prop_assert_eq!(g % d, 0);
            } else {
                // g == 0 only when both inputs are 0.
                prop_assert_eq!(a, 0);
                prop_assert_eq!(b, 0);
            }
        }

        // Edge case: gcd(a, 0) = a. The while-loop body never executes,
        // so this pins down the return value when b starts at 0.
        #[test]
        fn gcd_with_zero_second_arg_is_first_arg(a: u64) {
            prop_assert_eq!(greatest_common_divisor(a, 0), a);
        }

        // Edge case: gcd(0, b) = b. The loop performs a single swap
        // iteration; this pins down that behaviour.
        #[test]
        fn gcd_with_zero_first_arg_is_second_arg(b: u64) {
            prop_assert_eq!(greatest_common_divisor(0, b), b);
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    #[test]
    fn known() {
        assert_eq!(largest_smallest_integers(&[2, 4, 1, 3, 5, 7]), (None, Some(1)));
        assert_eq!(largest_smallest_integers(&[]), (None, None));
        assert_eq!(largest_smallest_integers(&[0, 0]), (None, None));
        assert_eq!(largest_smallest_integers(&[-3, -2, -5, 6, 7]), (Some(-2), Some(6)));
    }

    proptest! {
        // ---- First component: largest negative ----

        /// The first component is `Some` exactly when the list contains a
        /// negative element. Captures the "None when no such element" half
        /// of the contract.
        #[test]
        fn first_some_iff_has_negative(
            v in proptest::collection::vec(-100i64..100, 0..20)
        ) {
            let (a, _) = largest_smallest_integers(&v);
            let has_neg = v.iter().any(|&x| x < 0);
            prop_assert_eq!(a.is_some(), has_neg);
        }

        /// When the first component is `Some(x)`, then `x` is negative,
        /// `x` belongs to the list, and `x` is the maximum among all
        /// negative elements (i.e. the "largest negative" claim).
        #[test]
        fn first_is_largest_negative(
            v in proptest::collection::vec(-100i64..100, 0..20)
        ) {
            let (a, _) = largest_smallest_integers(&v);
            if let Some(x) = a {
                prop_assert!(x < 0);
                prop_assert!(v.contains(&x));
                for &y in &v {
                    if y < 0 {
                        prop_assert!(y <= x);
                    }
                }
            }
        }

        // ---- Second component: smallest positive ----

        /// The second component is `Some` exactly when the list contains a
        /// positive element. Captures the "None when no such element" half
        /// of the contract.
        #[test]
        fn second_some_iff_has_positive(
            v in proptest::collection::vec(-100i64..100, 0..20)
        ) {
            let (_, b) = largest_smallest_integers(&v);
            let has_pos = v.iter().any(|&x| x > 0);
            prop_assert_eq!(b.is_some(), has_pos);
        }

        /// When the second component is `Some(y)`, then `y` is positive,
        /// `y` belongs to the list, and `y` is the minimum among all
        /// positive elements (i.e. the "smallest positive" claim).
        #[test]
        fn second_is_smallest_positive(
            v in proptest::collection::vec(-100i64..100, 0..20)
        ) {
            let (_, b) = largest_smallest_integers(&v);
            if let Some(y) = b {
                prop_assert!(y > 0);
                prop_assert!(v.contains(&y));
                for &z in &v {
                    if z > 0 {
                        prop_assert!(y <= z);
                    }
                }
            }
        }
    }
}

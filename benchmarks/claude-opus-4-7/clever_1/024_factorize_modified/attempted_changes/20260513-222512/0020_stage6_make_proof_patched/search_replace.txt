/-- Reducing `p *? p` (a `RustM i64`) to a pure result, under the
    no-overflow envelope. Mirrors `largest_divisor_modified`'s `h_sub`
    block but for multiplication. -/
private theorem p_mul_p_pure (p : i64)
    (h_p_nn : (0 : Int) ≤ p.toInt)
    (h_p_ub : p.toInt ≤ 2^31 - 1) :
    (p *? p : RustM i64) = pure (p * p) := by
  show (rust_primitives.ops.arith.Mul.mul p p : RustM i64) = pure (p * p)
  show (if BitVec.smulOverflow p.toBitVec p.toBitVec then
          (.fail .integerOverflow : RustM i64)
        else pure (p * p)) = pure (p * p)
  have h_no_ov : ¬ Int64.mulOverflow p p := p_mul_p_no_overflow p h_p_nn h_p_ub
  have h_no_bv : BitVec.smulOverflow p.toBitVec p.toBitVec = false := by
    simpa [Int64.mulOverflow] using h_no_ov
  rw [h_no_bv]; rfl

/-- `p + 1` doesn't overflow when `p + 1 ≤ 2^31` (with `p ≥ 2`). The
    sum is well within `[-2^63, 2^63)`. -/
private theorem p_add_one_no_overflow (p : i64)
    (h_p_ge_0 : (0 : Int) ≤ p.toInt)
    (h_p1_ub : p.toInt + 1 ≤ 2^31) :
    ¬ Int64.addOverflow p (1 : i64) := by
  rw [Int64.addOverflow_iff]
  rw [int64_toInt_one]
  intro h
  have h_const_pos : (2^31 : Int) ≤ (2 : Int)^(64 - 1) := by decide
  have h_const_neg : (-(2 : Int)^(64 - 1)) ≤ 0 := by decide
  rcases h with h | h
  · omega
  · omega

/-- Reduce `p +? 1` to `pure (p + 1)` under the no-overflow envelope. -/
private theorem p_add_one_pure (p : i64)
    (h_p_ge_0 : (0 : Int) ≤ p.toInt)
    (h_p1_ub : p.toInt + 1 ≤ 2^31) :
    (p +? (1 : i64) : RustM i64) = pure (p + 1) := by
  show (rust_primitives.ops.arith.Add.add p (1 : i64) : RustM i64) = pure (p + 1)
  show (if BitVec.saddOverflow p.toBitVec ((1 : i64).toBitVec) then
          (.fail .integerOverflow : RustM i64)
        else pure (p + 1)) = pure (p + 1)
  have h_no_ov : ¬ Int64.addOverflow p (1 : i64) :=
    p_add_one_no_overflow p h_p_ge_0 h_p1_ub
  have h_no_bv : BitVec.saddOverflow p.toBitVec ((1 : i64).toBitVec) = false := by
    simpa [Int64.addOverflow] using h_no_ov
  rw [h_no_bv]; rfl

/-- `(p + 1).toInt = p.toInt + 1` under the no-overflow envelope. -/
private theorem p_add_one_toInt (p : i64)
    (h_p_ge_0 : (0 : Int) ≤ p.toInt)
    (h_p1_ub : p.toInt + 1 ≤ 2^31) :
    (p + 1).toInt = p.toInt + 1 := by
  have h_no_ov : ¬ Int64.addOverflow p (1 : i64) :=
    p_add_one_no_overflow p h_p_ge_0 h_p1_ub
  rw [Int64.toInt_add_of_not_addOverflow h_no_ov, int64_toInt_one]

/-- For the algorithmic regime `2 ≤ p`, `p ≠ 0`. -/
private theorem p_ne_zero (p : i64) (h_p_ge_2 : 2 ≤ p.toInt) :
    p ≠ (0 : i64) := by
  intro h
  have : p.toInt = (0 : i64).toInt := by rw [h]
  rw [int64_toInt_zero] at this
  omega

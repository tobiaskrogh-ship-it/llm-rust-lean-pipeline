private theorem factorize_at_correct
    (n p : i64) (acc : alloc.vec.Vec i64 alloc.alloc.Global)
    (h_n_ge_1 : 1 ≤ n.toInt)
    (h_p_ge_2 : 2 ≤ p.toInt)
    (h_p_le_n : p.toInt ≤ n.toInt)
    (h_n_bound : n.toInt + 1 ≤ 2^31)
    (h_acc_prime : ∀ (j : Nat) (hj : j < acc.val.size),
                      IsPrime (acc.val[j]'hj).toInt)
    (h_acc_le_p : ∀ (j : Nat) (hj : j < acc.val.size),
                      (acc.val[j]'hj).toInt ≤ p.toInt)
    (h_acc_sorted : ∀ (j : Nat) (h1 : j < acc.val.size) (h2 : j+1 < acc.val.size),
                       (acc.val[j]'h1).toInt ≤ (acc.val[j+1]'h2).toInt)
    (h_frontier : ∀ q : Int, 2 ≤ q → q < p.toInt → ¬ q ∣ n.toInt) :
    ∃ v : alloc.vec.Vec i64 alloc.alloc.Global,
      clever_024_factorize.factorize_at n p acc = RustM.ok v ∧
      array_product_int v.val = array_product_int acc.val * n.toInt ∧
      (∀ (j : Nat) (hj : j < v.val.size), IsPrime (v.val[j]'hj).toInt) ∧
      (∀ (j : Nat) (h1 : j < v.val.size) (h2 : j+1 < v.val.size),
         (v.val[j]'h1).toInt ≤ (v.val[j+1]'h2).toInt) := by
  unfold clever_024_factorize.factorize_at
  by_cases h_n_le_1 : n ≤ (1 : i64)
  · -- BASE CASE: n ≤ 1. By h_n_ge_1 (n.toInt ≥ 1) and h_n_le_1 (n.toInt ≤ 1),
    -- n.toInt = 1, so `factorize_at` returns `acc` and the product invariant
    -- degenerates correctly because `product acc * 1 = product acc`.
    have h_n_eq_1 : n.toInt = 1 := by
      have hh := Int64.le_iff_toInt_le.mp h_n_le_1
      rw [int64_toInt_one] at hh; omega
    have h_dec : decide (n ≤ (1 : i64)) = true := decide_eq_true h_n_le_1
    simp only [show (n <=? (1 : i64)) =
                 (pure (decide (n ≤ (1 : i64))) : RustM Bool) from rfl,
               h_dec, pure_bind, ↓reduceIte]
    -- Goal: ∃ v, pure acc = ok v ∧ ... — pick v := acc.
    refine ⟨acc, rfl, ?_, h_acc_prime, h_acc_sorted⟩
    -- Product invariant: array_product_int acc.val = array_product_int acc.val * n.toInt
    rw [h_n_eq_1]
    -- Now: array_product_int acc.val = array_product_int acc.val * 1
    rw [Int.mul_one]
  · -- RECURSIVE CASE: n > 1. Three sub-branches inside the function body:
    --   (a) `p *? p > n`         → push `n` (which is prime under the frontier
    --                              invariant + p² > n), return `acc ++ [n]`.
    --   (b) `n %? p == 0`        → push `p` (prime under frontier), recurse
    --                              on `(n/p, p, acc ++ [p])`.
    --   (c) otherwise            → recurse on `(n, p+1, acc)` with extended
    --                              frontier.
    --
    -- STUCK SUB-GOAL: even setting up the strong induction is blocked by the
    -- need for a *combined* measure. Branch (b) shrinks `n.toInt.toNat` by a
    -- factor of `p ≥ 2`, branch (c) keeps `n.toInt.toNat` constant but
    -- increments `p.toInt.toNat`. Single-parameter `Nat.strongRecOn` on
    -- `n.toInt.toNat` — the pattern from `gcd_recursive_modified` — does not
    -- decrease in branch (c), so the IH does not apply there. The natural
    -- measure is `(n.toInt.toNat, 2^31 - p.toInt.toNat)` lex-combined as
    -- `n.toInt.toNat * 2^31 + (2^31 - p.toInt.toNat)` (bounded by
    -- `h_n_bound`), but each branch's measure-decrease lemma must itself be
    -- proved (e.g. `(n / p).toInt.toNat * 2^31 + 2^31 < n.toInt.toNat * 2^31
    --   + (2^31 - p.toInt.toNat)` when `p ≥ 2` divides `n`).
    --
    -- Beyond the induction setup, each branch requires:
    --   * overflow-free discharge of `p *? p`, `n %? p`, `n /? p`, `p +? 1`
    --     (the partial-operator chains, c.f. `largest_divisor_modified`);
    --   * Vec push-step (mirror `rolling_max_at_step` /
    --     `shift_at_step` for `extend_from_slice` with a typed `[i64; 1]`
    --     chunk);
    --   * propagation of `acc_prime` / `acc_le_p` / `acc_sorted` when
    --     extending by `p` or `n`;
    --   * an `IsPrime`-of-trial-division lemma:
    --       `∀ q, 2 ≤ q → q < p → ¬ q ∣ n` ∧ `p ≥ 2` ∧ `p ∣ n` ⟹ `IsPrime p`
    --     and
    --       `∀ q, 2 ≤ q → q < p → ¬ q ∣ n` ∧ `2 ≤ n` ∧ `p*p > n` ⟹ `IsPrime n`
    --     (smallest-divisor / sqrt-bound primality).
    --
    -- See the structural-unblock docstring on `factorize_at_correct` itself
    -- for the full picture; see also the consumer theorems below, all of
    -- which derive in two lines once this helper closes.
    sorry

fn factorize_at(n: i64, p: i64, mut acc: Vec<i64>) -> Vec<i64> {
    if n <= 1 {
        acc
    } else if p * p > n {
        // Typed-chunk `extend_from_slice` rewrite per
        // `rewrite_patterns/vec_push_to_extend_from_slice_typed_chunk.rs`:
        // `Vec::push` has no Hax Lean prelude model, but
        // `Vec::extend_from_slice` does, provided the slice's element-count
        // is fixed in the let binding's type so `unsize` can elaborate it.
        let chunk: [i64; 1] = [n];
        acc.extend_from_slice(&chunk);
        acc
    } else if n % p == 0 {
        let chunk: [i64; 1] = [p];
        acc.extend_from_slice(&chunk);
        factorize_at(n / p, p, acc)
    } else {
        factorize_at(n, p + 1, acc)
    }
}

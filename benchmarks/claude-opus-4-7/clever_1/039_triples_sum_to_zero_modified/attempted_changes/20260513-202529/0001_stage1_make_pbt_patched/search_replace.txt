pub fn triples_sum_to_zero(numbers: &[u64]) -> bool {
    count_zeros_at(numbers, 0, 0) >= 3
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    proptest! {
        /// Completeness (postcondition, "true" side of the iff):
        /// when three *distinct* positions of the slice hold `u64` values
        /// that sum to zero — which, over `u64`, forces all three to be
        /// zero — the function must return true. We construct such a
        /// slice by interleaving three zeros at non-adjacent positions
        /// among arbitrary other values, so the test exercises the
        /// "distinct indices" part of the contract, not just "three
        /// zeros in a row".
        #[test]
        fn prop_returns_true_when_three_zero_positions_exist(
            a in proptest::collection::vec(any::<u64>(), 0..15),
            b in proptest::collection::vec(any::<u64>(), 0..15),
            c in proptest::collection::vec(any::<u64>(), 0..15),
            d in proptest::collection::vec(any::<u64>(), 0..15),
        ) {
            // xs = a ++ [0] ++ b ++ [0] ++ c ++ [0] ++ d
            // — three zeros at three distinct indices, with arbitrary
            // (possibly non-zero, possibly zero) padding around them.
            let mut xs: Vec<u64> = Vec::new();
            xs.extend(a);
            xs.push(0);
            xs.extend(b);
            xs.push(0);
            xs.extend(c);
            xs.push(0);
            xs.extend(d);
            prop_assert!(triples_sum_to_zero(&xs));
        }

        /// Soundness (postcondition, "false" side of the iff):
        /// three `u64` values sum to zero only if all three are zero, so
        /// a slice that contains *fewer than three* zero entries admits
        /// no zero-sum triple of distinct indices, and the function must
        /// return false. This covers the edge cases of empty slices and
        /// slices shorter than 3 as the `num_zeros = 0` instance, and
        /// rules out "always return true" or off-by-one bugs in the
        /// counting threshold.
        #[test]
        fn prop_returns_false_when_fewer_than_three_zero_positions(
            nonzeros in proptest::collection::vec(1u64..=u64::MAX, 0..30),
            num_zeros in 0usize..=2,
        ) {
            let mut xs = nonzeros;
            for _ in 0..num_zeros {
                xs.push(0);
            }
            prop_assert!(!triples_sum_to_zero(&xs));
        }
    }
}

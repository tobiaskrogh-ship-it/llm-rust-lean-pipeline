private theorem insert_sorted_at_length (v : RustSlice i64) (x : i64) :
    ∀ (n : Nat) (i : usize) (inserted : Bool)
      (acc : alloc.vec.Vec i64 alloc.alloc.Global),
      v.val.size - i.toNat ≤ n →
      i.toNat ≤ v.val.size →
      acc.val.size + (v.val.size - i.toNat) +
        (if inserted then 0 else 1) < USize64.size →
      ∃ w : alloc.vec.Vec i64 alloc.alloc.Global,
        clever_036_sort_even.insert_sorted_at v x i inserted acc = RustM.ok w ∧
        w.val.size = acc.val.size + (v.val.size - i.toNat) +
                       (if inserted then 0 else 1) := by
  intro n
  induction n with
  | zero =>
    intro i inserted acc hm hi_le h_fits
    have hi_eq : i.toNat = v.val.size := by omega
    have hi_ge : v.val.size ≤ i.toNat := by omega
    -- Base case: i.toNat = v.val.size, so v.val.size - i.toNat = 0.
    have h_sub_zero : v.val.size - i.toNat = 0 := by omega
    cases inserted with
    | true =>
      -- Returns acc.
      refine ⟨acc, insert_sorted_at_oob_inserted v x i acc hi_ge, ?_⟩
      show acc.val.size = acc.val.size + (v.val.size - i.toNat) + 0
      rw [h_sub_zero]; omega
    | false =>
      -- Pushes x. Size = acc.size + 1.
      have h_acc_succ : acc.val.size + 1 < USize64.size := by
        have h_temp : acc.val.size + (v.val.size - i.toNat) + 1 < USize64.size := by
          have : (if false then 0 else 1) = 1 := rfl
          rw [this] at h_fits; exact h_fits
        rw [h_sub_zero] at h_temp; omega
      refine ⟨push_one acc x h_acc_succ,
              insert_sorted_at_oob_not_inserted v x i acc hi_ge h_acc_succ, ?_⟩
      show (push_one acc x h_acc_succ).val.size = acc.val.size + (v.val.size - i.toNat) + 1
      show (acc.val ++ #[x]).size = acc.val.size + (v.val.size - i.toNat) + 1
      rw [Array.size_append, h_sub_zero]
      show acc.val.size + #[x].size = acc.val.size + 0 + 1
      have : (#[x] : Array i64).size = 1 := rfl
      rw [this]; omega

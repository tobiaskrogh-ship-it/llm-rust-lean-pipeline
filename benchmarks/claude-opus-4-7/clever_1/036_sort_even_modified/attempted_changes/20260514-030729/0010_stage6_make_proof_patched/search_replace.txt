    `USize64.size` (always true for `RustSlice` indices). -/

/-- `usize` mod by `2`: since `2 ≠ 0`, the `Rem` succeeds with `pure (i % 2)`. -/
private theorem usize_mod_two_eq (i : usize) :
    (i %? (2 : usize) : RustM usize) = RustM.ok (i % 2 : usize) := by
  show (rust_primitives.ops.arith.Rem.rem i (2 : usize) : RustM usize) = _
  show (if (2 : usize) = 0 then (.fail .divisionByZero : RustM usize)
        else pure (i % (2 : usize))) = _
  have h2 : ((2 : usize) = 0) = False := by
    apply propext
    constructor
    · intro h; exact absurd h (by decide)
    · intro h; exact False.elim h
  rw [show ((2 : usize) = 0) ↔ False from by rw [h2]] at *
  simp only [if_false]
  rfl

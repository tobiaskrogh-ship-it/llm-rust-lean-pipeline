    have h_new : (alloc.vec.Impl.new i64 rust_primitives.hax.Tuple0.mk :
                    RustM (alloc.vec.Vec i64 alloc.alloc.Global)) =
                  RustM.ok v_empty := rfl

    have h_zero_toNat : (0 : usize).toNat = 0 := rfl

    rw [h_collect, RustM_ok_bind]

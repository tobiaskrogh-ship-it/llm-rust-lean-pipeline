private theorem scan_at_argmin_correct
    (numbers : RustSlice i64) (retI retJ : usize)
    (hlen : 2 ≤ numbers.val.size)
    (hscan : scan_at numbers (0 : usize) (1 : usize) (0 : usize) (1 : usize)
              = RustM.ok (rust_primitives.hax.Tuple2.mk retI retJ)) :
    ∃ (hretI : retI.toNat < numbers.val.size)
      (hretJ : retJ.toNat < numbers.val.size),
      retI.toNat ≠ retJ.toNat ∧
      ∀ (p q : Nat) (hp : p < numbers.val.size) (hq : q < numbers.val.size),
        p < q →
        abs_diff_int numbers retI.toNat retJ.toNat hretI hretJ ≤
          abs_diff_int numbers p q hp hq := by
  -- Distinctness part: closed using `scan_at_returns_valid_pair`. The seed
  -- (0, 1) has 0 < size, 1 < size (from hlen ≥ 2), and 0 ≠ 1.
  have h0_lt : (0 : usize).toNat < numbers.val.size := by
    show 0 < numbers.val.size; omega
  have h1_lt : (1 : usize).toNat < numbers.val.size := by
    show 1 < numbers.val.size; omega
  have h01_ne : (0 : usize).toNat ≠ (1 : usize).toNat := by decide
  obtain ⟨hretI, hretJ, hne⟩ :=
    scan_at_returns_valid_pair numbers (0 : usize) (1 : usize) (0 : usize) (1 : usize)
      retI retJ h0_lt h1_lt h01_ne hscan
  refine ⟨hretI, hretJ, hne, ?_⟩
  -- Minimality part: stuck sub-goal. After narrowing distinctness, we need:
  --   ∀ p < q < size, abs_diff_int numbers retI.toNat retJ.toNat ≤ abs_diff_int numbers p q
  -- This requires that `scan_at` actually minimised the abs-diff over all
  -- pairs examined by the recursion. Specifically, we'd need a strengthened
  -- inductive invariant `scan_at_argmin_inductive_invariant(numbers, i, j,
  -- bi, bj, retI, retJ)` whose conclusion is:
  --   (retI, retJ) is argmin of abs_diff over the lex-future of (i, j),
  --   union {(bi, bj)} as the current best.
  -- Then strong induction on the (size, i, j) measure closes via case-split
  -- on each branch of `scan_at`. The `abs_diff` overflow chase in the inner
  -- branch is the technically hardest piece (cf. Clever_000.step_analyze
  -- which is ~250 lines for a single-counter scan).
  intro p q hp hq hpq
  sorry

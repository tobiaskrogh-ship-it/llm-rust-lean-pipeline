/// Reads an unsigned 64 bit integer from `buf` in big-endian byte order.
///
/// # Panics
///
/// Panics when `buf.len() < 8`.
//
// Original body: `u64::from_be_bytes(buf[..8].try_into().unwrap())`.
// Four constructs in that one expression are outside the Hax-modeled
// fragment and were rewritten away:
//   1. `buf[..8]` range-slice indexing — Hax models single-element
//      indexing (the partial `_[_]?` operator) but not range slicing
//      into a fresh slice here.
//   2. `<&[u8] as TryInto<[u8; 8]>>::try_into` — the slice→array
//      `TryInto` conversion has no Hax prelude model.
//   3. `Result::unwrap` on the `TryInto` result.
//   4. `u64::from_be_bytes` — extracts to
//      `core_models.num.Impl_*.from_be_bytes`, undefined in the Hax
//      Lean prelude (`lake build` would fail with `Unknown identifier`).
// Replacement: assemble the value directly from the first eight bytes
// with shift/mask, the proven Hax-compatible shape for big-endian
// reads (cf. `big_endian_from_slice_u64_modified`, which uses only
// `<<`/`>>`/`&`/`|` over `u64`; single-element indexing `buf[i]` is the
// modeled partial operator, cf. `slice_get_u64_modified`; the `u8 as
// u64` widening cast is a modeled integer-to-integer cast). Panic
// semantics are preserved exactly: for `buf.len() < 8` the first
// out-of-range index (`buf[buf.len()]`, up to `buf[7]`) panics just as
// the original `buf[..8]` slice did; for `buf.len() == 8` every access
// is in bounds and no panic occurs; bytes at index >= 8 are never read.
#[inline]
pub fn read_u64(buf: &[u8]) -> u64 {
    ((buf[0] as u64) << 56)
        | ((buf[1] as u64) << 48)
        | ((buf[2] as u64) << 40)
        | ((buf[3] as u64) << 32)
        | ((buf[4] as u64) << 24)
        | ((buf[5] as u64) << 16)
        | ((buf[6] as u64) << 8)
        | (buf[7] as u64)
}

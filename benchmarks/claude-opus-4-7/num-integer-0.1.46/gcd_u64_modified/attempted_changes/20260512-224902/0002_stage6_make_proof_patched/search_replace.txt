theorem gcd_is_greatest (x y d : u64) :
    ⦃ ⌜ d.toNat ∣ x.toNat ∧ d.toNat ∣ y.toNat ⌝ ⦄
      gcd_u64.gcd x y
    ⦃ ⇓ g => ⌜ d.toNat ∣ g.toNat ⌝ ⦄ := by
  rw [RustM.Triple_iff_BitVec, gcd_characterization]
  by_cases h : d.toNat ∣ x.toNat ∧ d.toNat ∣ y.toNat
  · -- Precondition holds: the LHS `!decide (pre)` is `false`, so the
    -- `||` reduces to the second clause. Reduce and apply `Nat.dvd_gcd`.
    simp only [h, decide_true, Bool.not_true, Bool.false_or,
               RustM.toBVRustM_ok, Bool.true_and, decide_eq_true_eq]
    rw [gcd_toNat_ofNat]
    exact Nat.dvd_gcd h.1 h.2
  · -- Precondition fails: the `||` short-circuits via the negated LHS.
    simp only [h, decide_false, Bool.not_false, Bool.true_or]

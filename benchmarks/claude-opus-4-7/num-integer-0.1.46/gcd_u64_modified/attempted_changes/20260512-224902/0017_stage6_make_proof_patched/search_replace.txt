/-- `toNat` of the `u64` cast of `Nat.gcd` is the `Nat.gcd` itself, with

private theorem gcd_toNat_ofNat (a b : u64) :

/-! ## Body-step scaffolding for the strip-2 loops (L2/L3/L5/L6)

A reusable u64-level body step that the strip-2-from-m loop body
discharge can invoke directly. States:
  When `m & 1 = 0` and the odd-fixed coprime invariant holds for `m`
  and a fixed odd `n`, the body action `m ← m >>> 1` (a) does not
  overflow (shift-right is always safe), (b) strictly decreases
  `m.toNat`, and (c) preserves `Nat.gcd m.toNat n.toNat`.

This is the Nat-level content the L2 body-step Hoare triple needs;
the Hoare-triple wrapping is structurally identical to the
`body_step_nat` of `while_example/While_exampleObligations.lean`. -/
private theorem strip_two_body_step_nat
    (m n : u64) (hm_pos : m.toNat > 0) (hm_even : m.toNat % 2 = 0)
    (hn_odd : n.toNat % 2 = 1) :
    -- (a) shift-right decreases (toNat-level)
    (m >>> (1 : i32)).toNat < m.toNat ∧
    -- (b) shift-right keeps it positive
    (m >>> (1 : i32)).toNat > 0 ∧
    -- (c) gcd preserved
    Nat.gcd (m >>> (1 : i32)).toNat n.toNat = Nat.gcd m.toNat n.toNat := by
  -- Reduce `(m >>> 1).toNat` to `m.toNat / 2` via `UInt64.toNat_shiftRight`.
  have h_shift : (m >>> (1 : i32)).toNat = m.toNat / 2 := by
    -- `(m >>> i).toNat = m.toNat >>> (i.toNat % 64)` (approximately).
    -- For `i = (1 : i32)`, this is `m.toNat / 2`.
    have h := UInt64.toNat_shiftRight m (1 : i32)
    -- h : (m >>> 1).toNat = m.toNat >>> ((1 : i32).toNat % 64)
    -- Simplify via `Nat.shiftRight_eq_div_pow` and `(1 : i32).toNat = 1`.
    rw [h]
    -- Goal: m.toNat >>> ((1 : i32).toNat % 64) = m.toNat / 2
    -- `(1 : i32).toNat` should be 1. `1 % 64 = 1`. `m >>> 1 = m / 2^1 = m/2`.
    simp [Nat.shiftRight_eq_div_pow]
  refine ⟨?_, ?_, ?_⟩
  · rw [h_shift]; omega
  · rw [h_shift]
    -- m / 2 > 0 needs m ≥ 2. From m > 0 and m even, m ≥ 2.
    omega
  · rw [h_shift]
    exact (nat_gcd_div_two_left_odd_right m.toNat n.toNat hm_even hn_odd).symm

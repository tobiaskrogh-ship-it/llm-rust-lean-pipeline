/-- Subtraction step: when `b ≤ a`, `gcd a b = gcd (a - b) b`.

    Direct two-side `dvd` argument via `Nat.dvd_sub` and `Nat.dvd_add`
    on the cancellation `(a - b) + b = a`. -/
private theorem nat_gcd_sub_left (a b : Nat) (h : b ≤ a) :
    Nat.gcd a b = Nat.gcd (a - b) b := by
  apply Nat.dvd_antisymm
  · apply Nat.dvd_gcd
    · exact Nat.dvd_sub (Nat.gcd_dvd_left a b) (Nat.gcd_dvd_right a b)
    · exact Nat.gcd_dvd_right a b
  · apply Nat.dvd_gcd
    · have h1 : Nat.gcd (a - b) b ∣ (a - b) := Nat.gcd_dvd_left _ _
      have h2 : Nat.gcd (a - b) b ∣ b := Nat.gcd_dvd_right _ _
      have hsum : Nat.gcd (a - b) b ∣ ((a - b) + b) := Nat.dvd_add h1 h2
      rw [Nat.sub_add_cancel h] at hsum
      exact hsum
    · exact Nat.gcd_dvd_right _ _

/-- Subtraction step: when `b ≤ a`, `gcd a b = gcd (a - b) b`.

    Direct two-side `dvd` argument via `Nat.dvd_sub` and `Nat.dvd_add`
    on the cancellation `(a - b) + b = a`. -/
private theorem nat_gcd_sub_left (a b : Nat) (h : b ≤ a) :
    Nat.gcd a b = Nat.gcd (a - b) b := by
  apply Nat.dvd_antisymm
  · apply Nat.dvd_gcd
    · exact Nat.dvd_sub (Nat.gcd_dvd_left a b) (Nat.gcd_dvd_right a b)
    · exact Nat.gcd_dvd_right a b
  · apply Nat.dvd_gcd
    · have h1 : Nat.gcd (a - b) b ∣ (a - b) := Nat.gcd_dvd_left _ _
      have h2 : Nat.gcd (a - b) b ∣ b := Nat.gcd_dvd_right _ _
      have hsum : Nat.gcd (a - b) b ∣ ((a - b) + b) := Nat.dvd_add h1 h2
      rw [Nat.sub_add_cancel h] at hsum
      exact hsum
    · exact Nat.gcd_dvd_right _ _

/-- "One even, one odd" gcd reduction.

    When `a` is even (say `a = 2 * a'`) and `b` is odd, the factor 2 in
    `a` does not divide `b`, so it cannot appear in `gcd a b`. Hence
    halving `a` does not lose any factor.

    Concretely: under the hypothesis `b % 2 = 1`, we have
    `Nat.gcd (2 * a) b = Nat.gcd a b`.

    Proof structure: two-side dvd argument via `Nat.dvd_antisymm`.
      * `gcd a b ∣ a ∣ 2*a`, and `gcd a b ∣ b`, so `gcd a b ∣ gcd (2*a) b`.
      * For the other direction, we use that any divisor of `2*a` that
        also divides an odd `b` must be odd, hence coprime with 2, hence
        divides `a`. This uses `Nat.Coprime.dvd_of_dvd_mul_left`. -/
private theorem nat_gcd_two_mul_left_odd_right (a b : Nat) (hb : b % 2 = 1) :
    Nat.gcd (2 * a) b = Nat.gcd a b := by
  apply Nat.dvd_antisymm
  · -- gcd(2a, b) ∣ gcd(a, b).
    -- Strategy: show gcd(2a, b) is coprime with 2 (since it divides odd b),
    -- then use coprime-cancellation on `gcd(2a, b) ∣ 2 * a`.
    have hg_dvd_2a : Nat.gcd (2 * a) b ∣ 2 * a := Nat.gcd_dvd_left _ _
    have hg_dvd_b : Nat.gcd (2 * a) b ∣ b := Nat.gcd_dvd_right _ _
    -- gcd(2a, b) is coprime with 2: any common divisor with 2 also divides b,
    -- but odd b cannot share an even factor. Concretely `Nat.Coprime g 2`.
    have hg_coprime_2 : Nat.Coprime (Nat.gcd (2 * a) b) 2 := by
      unfold Nat.Coprime
      -- gcd(gcd(2a,b), 2) ∣ 2, so it's 1 or 2. If 2, then 2 ∣ gcd(2a,b) ∣ b,
      -- contradicting b odd.
      rcases Nat.dvd_prime Nat.prime_two |>.mp (Nat.gcd_dvd_right (Nat.gcd (2*a) b) 2) with h1 | h2
      · exact h1
      · exfalso
        -- h2 : Nat.gcd (Nat.gcd (2*a) b) 2 = 2, so 2 ∣ gcd(2a, b) ∣ b
        have h2_dvd_g : (2 : Nat) ∣ Nat.gcd (2 * a) b := by
          have : Nat.gcd (Nat.gcd (2*a) b) 2 ∣ Nat.gcd (2*a) b := Nat.gcd_dvd_left _ _
          rw [h2] at this; exact this
        have h2_dvd_b : (2 : Nat) ∣ b := h2_dvd_g.trans hg_dvd_b
        -- 2 ∣ b contradicts b % 2 = 1
        rw [Nat.dvd_iff_mod_eq_zero] at h2_dvd_b
        omega
    -- gcd(2a, b) ∣ 2 * a and coprime with 2 ⇒ ∣ a.
    have hg_dvd_a : Nat.gcd (2 * a) b ∣ a := by
      exact (Nat.Coprime.dvd_of_dvd_mul_left hg_coprime_2 hg_dvd_2a)
    exact Nat.dvd_gcd hg_dvd_a hg_dvd_b
  · -- gcd(a, b) ∣ gcd(2a, b): trivial since gcd(a,b) ∣ a ⇒ ∣ 2a.
    apply Nat.dvd_gcd
    · exact (Nat.gcd_dvd_left a b).mul_left 2
    · exact Nat.gcd_dvd_right a b

/-- Strip-a-2 step on the left, when the right argument is odd.

    `Nat.gcd m n = Nat.gcd (m / 2) n` when `m % 2 = 0` and `n % 2 = 1`.
    Direct consequence of `nat_gcd_two_mul_left_odd_right` applied to
    `a = m / 2`, using `m = 2 * (m / 2)` for even `m`. -/
private theorem nat_gcd_div_two_left_odd_right
    (m n : Nat) (hm : m % 2 = 0) (hn : n % 2 = 1) :
    Nat.gcd m n = Nat.gcd (m / 2) n := by
  have hm2 : 2 * (m / 2) = m := by
    have := Nat.div_add_mod m 2
    omega
  conv_lhs => rw [← hm2]
  exact nat_gcd_two_mul_left_odd_right (m / 2) n hn

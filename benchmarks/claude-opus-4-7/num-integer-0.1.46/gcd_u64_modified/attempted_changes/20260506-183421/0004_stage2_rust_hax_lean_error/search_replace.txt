    while m != n {
        // Termination measure: bitwise OR `m | n` strictly decreases each
        // iteration.  In the `m > n` branch, m_new = (m - n) >> k for some
        // k >= 1 (since m - n is even — both odd before), so
        //   m_new < m - n < m   ⇒   m_new + n < m ≤ m | n
        // and `m_new | n_new <= m_new + n < m | n`.  The `m < n` branch is
        // symmetric.  Bitwise OR has no overflow check at extraction time,
        // unlike `m + n`, which trips Hax's `pureP` synthesis on u64
        // (`UInt64.addOverflow` may be true).
        hax_lib::loop_decreases!(m | n);

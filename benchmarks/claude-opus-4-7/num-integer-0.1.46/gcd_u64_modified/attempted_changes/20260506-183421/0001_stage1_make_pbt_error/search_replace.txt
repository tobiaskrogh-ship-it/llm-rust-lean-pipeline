    #[test]
    fn test_gcd_cmp_with_euclidean() {
        fn euclidean_gcd(mut m: u64, mut n: u64) -> u64 {

                mem::swap(&mut m, &mut n);
                m %= n;

            n
        }

        }

    // ---------- Property-based tests capturing the contract ----------
    //
    // Contract for `gcd: (u64, u64) -> u64`:
    //   * Preconditions: none (every (u64, u64) is valid input).
    //   * Failure conditions: none (the function is total; no panics).
    //   * Postconditions:
    //       (Z) Zero boundary: gcd(0, 0) = 0, gcd(x, 0) = gcd(0, x) = x.
    //       (D) The result divides both inputs (it is a common divisor).
    //       (G) The result is the greatest such divisor: every common
    //           divisor of the inputs divides the result.
    // (Z), (D), (G) together uniquely characterise the GCD on u64.

    /// Postcondition (Z): zero boundary cases.
    ///
    /// Distinguishes the convention `gcd(0, 0) = 0` and the identity
    /// behaviour `gcd(x, 0) = gcd(0, x) = x`, which the implementation
    /// realises via `m | n` on the early-return path.
    #[test]
    fn prop_gcd_zero_cases() {
        // gcd(0, 0) = 0
        assert_eq!(gcd(0, 0), 0);
        // gcd(x, 0) = x and gcd(0, x) = x
        for x in 0..=255u64 {
            assert_eq!(gcd(x, 0), x);
            assert_eq!(gcd(0, x), x);
        }
        // Spot-check the upper end of the range to guard against
        // surprises in `m | n` on large values.
        assert_eq!(gcd(u64::MAX, 0), u64::MAX);
        assert_eq!(gcd(0, u64::MAX), u64::MAX);
    }

    /// Postcondition (D): `gcd(x, y)` is a common divisor of `x` and `y`.
    ///
    /// When `g == 0` we additionally require `x == 0 && y == 0`, since
    /// otherwise `g` would not be defined as a divisor of a nonzero value.
    #[test]
    fn prop_gcd_divides_both() {
        for x in 0..=128u64 {
            for y in 0..=128u64 {
                let g = gcd(x, y);
                if g == 0 {
                    // The only way `g` can be 0 is the (0, 0) case.
                    assert_eq!(x, 0);
                    assert_eq!(y, 0);
                } else {
                    assert_eq!(x % g, 0, "gcd({}, {}) = {} does not divide x", x, y, g);
                    assert_eq!(y % g, 0, "gcd({}, {}) = {} does not divide y", x, y, g);
                }
            }
        }
    }

    /// Postcondition (G): every common divisor of `x` and `y` divides
    /// `gcd(x, y)`.
    ///
    /// This is independent of (D): an implementation that always returned
    /// `1` would satisfy (D) but fail (G). Conversely, an implementation
    /// that returned `lcm(x, y)` (or any multiple of the true GCD) could
    /// satisfy (G) but fail (D).
    #[test]
    fn prop_gcd_is_greatest() {
        for x in 0..=64u64 {
            for y in 0..=64u64 {
                let g = gcd(x, y);
                // Any common divisor `d` lies in 1..=max(x, y) (or is
                // arbitrary if both are 0; the bound 64 still covers
                // every divisor of x and y in this range).
                for d in 1..=64u64 {
                    if x % d == 0 && y % d == 0 {
                        // `d` is a common divisor, so it must divide `g`.
                        // When g == 0 we have x = y = 0 and 0 % d == 0,
                        // which is consistent with the convention.
                        assert_eq!(
                            g % d,
                            0,
                            "common divisor d = {} of x = {}, y = {} does not divide gcd = {}",
                            d, x, y, g,
                        );
                    }
                }
            }
        }
    }

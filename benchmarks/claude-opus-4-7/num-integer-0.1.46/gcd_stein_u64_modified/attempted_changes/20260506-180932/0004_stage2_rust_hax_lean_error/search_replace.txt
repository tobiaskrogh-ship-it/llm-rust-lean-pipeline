    while m != n {
        // Each outer iteration subtracts the smaller from the larger,
        // so `m + n` strictly decreases. We compute the measure in
        // u128 so it cannot overflow (the proof obligation
        // `_pureTermination` would fail if the u64 add could panic).
        // (`if-then-else` measures are also rejected by
        // `hax_construct_pure`: "mvcgen generated more than one goal
        // containing the metavariable.")
        hax_lib::loop_decreases!((m as u128) + (n as u128));

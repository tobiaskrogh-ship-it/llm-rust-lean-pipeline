    /// Cross-check with Euclidean gcd (the `test_gcd_cmp_with_euclidean`
    /// pattern from num-integer's test suite, src/lib.rs:984-998).
    #[test]
    fn matches_euclidean() {
        fn euclidean(mut m: u64, mut n: u64) -> u64 {
            while m != 0 {
                let t = m;
                m = n % m;
                n = t;
            }
            n
        }
        for a in 0u64..=50 {
            for b in 0u64..=50 {
                assert_eq!(gcd_stein(a, b), euclidean(a, b),
                    "gcd_stein({a}, {b}) disagrees with Euclidean");
            }
        }
    }

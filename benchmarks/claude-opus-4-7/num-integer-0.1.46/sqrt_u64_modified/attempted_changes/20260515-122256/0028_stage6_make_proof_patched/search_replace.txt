/-! ### log2 closed form (the explicit structural-unblock helper)

The `sqrt_u64.sqrt` proof requires a closed-form spec for the helper
`sqrt_u64.log2`. Mirror `Gcd_whileObligations.gcd_while_postcondition` exactly:

  * State the loop invariant `y.toNat = x.toNat / 2 ^ count.toNat ∧ count.toNat ≤ 63`.
  * Pure cond: `decide (y.toNat > 1)`.
  * Termination measure: `y.toNat`.
  * Body discharges via `UInt64.toNat_shiftRight` + `UInt32.toNat_add_of_lt`.

Stated below as a private theorem with a substantive `unfold + simp_only`
opening tactic; the remaining Stage-1 Hoare-triple body is the structural
unblock for `sqrt_postcondition`. -/

/-- Closed-form spec for the `log2` helper. For non-zero `x`, returns
    `⌊log₂ x.toNat⌋`. For `x = 0`, returns `0` (the while-loop exits
    immediately since `0 > 1` is false). -/
private theorem log2_postcondition_aux (x : u64) :
    sqrt_u64.log2 x = RustM.ok (UInt32.ofNat (Nat.log2 x.toNat)) := by
  -- Substantive attempt: open up the loop. Mirrors `gcd_while_postcondition`'s
  -- Stage-1+2 pattern (see `Gcd_whileObligations.lean:160`).
  unfold sqrt_u64.log2
  unfold rust_primitives.hax.while_loop
  -- After unfolding, the body should be
  --   `Lean.Loop.MonoLoopCombinator.while_loop ⟨0, x⟩ body >>= fun ⟨c,_⟩ => pure c`.
  -- The Stage-1 step is to instantiate
  -- `Spec.MonoLoopCombinator.while_loop` with invariant
  --   `inv ⟨count, y⟩ := y.toNat = x.toNat / 2 ^ count.toNat ∧ count.toNat ≤ 63`
  -- and termination `term ⟨_, y⟩ := y.toNat`, then discharge the body step
  -- via `UInt64.toNat_shiftRight` (`(y >>> 1).toNat = y.toNat / 2`) and
  -- `UInt32.toNat_add_of_lt` (count+1 doesn't overflow u32).
  --
  -- Stuck sub-goal: at this point the proof state is the full Hoare-triple
  -- equation for the bound-form
  --   `(LoopComb.while_loop ⟨0,x⟩ body >>= pure ∘ Tuple2.fst) =
  --    RustM.ok (UInt32.ofNat (Nat.log2 x.toNat))`
  -- which is what `RustM.Triple_iff_BitVec` converts the Stage-1 triple
  -- into. The body-step Nat-level facts are mechanical but the iota
  -- reduction on the destructured `⟨count, y⟩` pattern is fragile and
  -- needs the `cases s with | mk count y` style used in
  -- `gcd_loop_triple` (`Gcd_whileObligations.lean:73-103`).
  --
  -- Structural unblock: an explicit `body_step_nat` helper proving
  --   ∀ count y, count.toNat ≤ 62 → y.toNat = x.toNat / 2^count.toNat →
  --     y.toNat > 1 →
  --       UInt32.addOverflow count 1 = false ∧
  --       (y >>> (1 : u64)).toNat / 2^(count.toNat+1) * 2^(count.toNat+1) ≤ x.toNat ∧
  --       (count + 1).toNat ≤ 63
  -- then port the Stage-1+2 plumbing verbatim from `gcd_while_postcondition`.
  sorry

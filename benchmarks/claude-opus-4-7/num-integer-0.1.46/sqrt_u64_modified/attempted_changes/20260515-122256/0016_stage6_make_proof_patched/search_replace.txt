private theorem nat_sqrt_unique (x r s : Nat)
    (hr_le : r * r ≤ x) (hr_lt : x < (r + 1) * (r + 1))
    (hs_le : s * s ≤ x) (hs_lt : x < (s + 1) * (s + 1)) :
    r = s := by
  -- Both directions: r ≤ s and s ≤ r.
  -- For r ≤ s: assume ¬(r ≤ s), then s + 1 ≤ r, so r*r ≥ (s+1)*(s+1) > x, contradiction.
  have hrs : r ≤ s := by
    by_contra hgt
    have hge : s + 1 ≤ r := Nat.lt_of_not_le hgt
    have h_sq_le_sq : (s + 1) * (s + 1) ≤ r * r :=
      Nat.mul_le_mul hge hge
    omega
  have hsr : s ≤ r := by
    by_contra hgt
    have hge : r + 1 ≤ s := Nat.lt_of_not_le hgt
    have h_sq_le_sq : (r + 1) * (r + 1) ≤ s * s :=
      Nat.mul_le_mul hge hge
    omega
  omega

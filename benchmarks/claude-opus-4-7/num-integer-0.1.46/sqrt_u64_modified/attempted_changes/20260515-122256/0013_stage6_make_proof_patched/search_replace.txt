theorem sqrt_four : sqrt_u64.sqrt 4 = RustM.ok 2 := by
  rw [sqrt_postcondition]
  congr 1
  apply UInt64.toNat_inj.mp
  rw [UInt64.toNat_ofNat_of_lt' (nat_sqrt_lt_2_64 4)]

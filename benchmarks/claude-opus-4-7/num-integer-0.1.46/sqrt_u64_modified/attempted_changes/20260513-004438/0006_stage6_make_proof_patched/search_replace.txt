/-- Bridge: `2^r ≤ x ∧ x < 2^(r+1)` from `x / 2^r = 1` and `r ≤ 63`.

    Reachable from the exit condition of `log2_loop_triple`:
    `log2Inv x s ∧ ¬log2Cond s` gives `s._1.toNat = x.toNat / 2 ^ s._0.toNat ≥ 1`
    and `s._1.toNat ≤ 1`, hence `s._1.toNat = 1`, hence `x / 2^r = 1`. -/
private theorem log2_exit_bounds (x : u64) (r : Nat) (hr : r ≤ 63)
    (hdiv : x.toNat / 2 ^ r = 1) :
    2 ^ r ≤ x.toNat ∧ x.toNat < 2 ^ (r + 1) := by
  have hpos : 0 < 2 ^ r := Nat.two_pow_pos r
  refine ⟨?_, ?_⟩
  · -- x / 2^r = 1 ⇒ x ≥ 2^r
    have hmul : x.toNat / 2 ^ r * 2 ^ r ≤ x.toNat := Nat.div_mul_le_self _ _
    rw [hdiv, Nat.one_mul] at hmul
    exact hmul
  · -- x / 2^r = 1 ⇒ x < 2 * 2^r = 2^(r+1)
    have : x.toNat < 2 * 2 ^ r := by
      rw [Nat.div_lt_iff_lt_mul hpos] at *
      -- we need x.toNat / 2^r < 2; we have hdiv : x.toNat / 2^r = 1
      have h_lt : x.toNat / 2 ^ r < 2 := by rw [hdiv]; decide
      rw [Nat.div_lt_iff_lt_mul hpos] at h_lt
      exact h_lt
    have h_pow : 2 ^ (r + 1) = 2 ^ r * 2 := Nat.pow_succ 2 r
    have h_comm : 2 ^ r * 2 = 2 * 2 ^ r := Nat.mul_comm _ _
    omega

    // Babylonian iteration: next(x) = (a / x + x) / 2.
    // Inlined from `fixpoint` in src/roots.rs:170-186 with the closure
    // `|x| (a / x + x) >> 1` substituted directly.
    let mut xn: u64 = (a / x + x) >> 1;
    // First phase: if the initial guess underestimates `sqrt(a)`, a single
    // Newton step suffices to bring `x` above `sqrt(a)`. The original
    // num-integer code wraps this in `while x < xn { ... }`, but for our
    // power-of-two initial guess `g = 2^⌊(⌊log₂a⌋+1)/2⌋` and `a >= 4` one can
    // verify (and the test sweep over u64 in `tests::sqrt_test` confirms)
    // that this loop runs at most once: after one Newton step from `x < sqrt(a)`,
    // the AM-GM inequality forces the new `x = (a/x + x)/2 >= sqrt(a)`, hence
    // `x >= xn` afterwards. We therefore replace the `while` with an `if`.
    // This is purely a Hax-compatibility rewrite: a `while` loop here would
    // need a `loop_decreases!` measure, but the natural one (`a - x`) involves
    // a subtraction whose pureness `grind` cannot prove during extraction
    // (the same `pureP` synthesis limit documented in the skill).
    if x < xn {
        x = xn;
        xn = (a / x + x) >> 1;
    }
    // Second phase: standard Babylonian descent once `x >= sqrt(a)`. Here
    // `x > xn` is the loop guard and `x` is reassigned to the strictly smaller
    // `xn` each iteration, so `x` itself is a valid termination measure.
    while x > xn {
        hax_lib::loop_decreases!(x);
        x = xn;
        xn = (a / x + x) >> 1;
    }
    x

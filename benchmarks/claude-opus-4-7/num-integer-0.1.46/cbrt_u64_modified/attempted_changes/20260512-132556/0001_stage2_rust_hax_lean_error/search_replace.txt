fn fixpoint_cbrt(a: u64, mut x: u64) -> u64 {

    while x < xn {
        x = xn;
        xn = (a / (x * x) + x * 2) / 3;
    }
    while x > xn {
        x = xn;
        xn = (a / (x * x) + x * 2) / 3;
    }
    x
}

/// Integer-only initial guess for the Newton fixpoint, replacing the
/// `(a as f64).cbrt() as u64` guess of the source. The fixpoint
/// converges to the same value regardless of the starting point so long
/// as `x * x` does not overflow `u64`.
///
/// Returns a power-of-two `g` with `cbrt(a) <= g < 2^32`, so that the
/// recurrence `(a / (x*x) + 2*x) / 3` stays inside `u64` from the very
/// first step. The `f64`-based guess used by the source is unavailable
/// here: `as f64` and `f64::cbrt` have no Hax / Lean model.
fn cbrt_guess_u64(a: u64) -> u64 {
    // floor(log2(a)) via shift loop. `a >= 8` at every call site, so
    // `hi >= 3` on exit and the loop runs at least three times.
    let mut hi: u32 = 0;
    let mut y: u64 = a;
    while y > 1 {
        y >>= 1;
        hi += 1;
    }
    // k = ceil((hi + 1) / 3); we have hi <= 63 here so k <= 22 < 32.
    let k: u32 = (hi + 3) / 3;
    // Compute 2^k via a doubling loop (avoids a runtime-amount shift).
    let mut g: u64 = 1;
    let mut i: u32 = 0;
    while i < k {
        g <<= 1;
        i += 1;
    }
    g
}

/// Concrete `u64` cube root --- truncated principal `∛x`.
///
/// `cbrt(x)` returns the largest `r: u64` with `r*r*r <= x`.
pub fn cbrt(x: u64) -> u64 {
    let a = x;

    // Source has `if a < 8 { return (a > 0) as $T; }` --- inlined.
    if a < 8 {
        return if a > 0 { 1 } else { 0 };
    }

    // Source has `if a <= core::u32::MAX as $T { return (a as u32).cbrt() as $T; }`.
    // The associated constant `u32::MAX` extracts to
    // `core_models.num.Impl_8.MAX`, which the Hax Lean prelude does not
    // define (lake build: `Unknown identifier 'core_models.num.Impl_8.MAX'`).
    // Inline the literal value `u32::MAX = 4_294_967_295`.
    if a <= 4_294_967_295u64 {
        return cbrt_u32(a as u32) as u64;
    }

    // Integer-only initial guess + Newton fixpoint. The source's
    // `(a as f64).cbrt() as u64` guess relies on `as f64` (no `Cast u64 f64`
    // instance in the Hax Lean prelude) and `f64::cbrt` (no
    // `std.f64.Impl.cbrt` identifier); both cause `lake build` errors.
    // The fixpoint itself is unchanged, so the final value is identical.
    let guess = cbrt_guess_u64(a);
    fixpoint_cbrt(a, guess)
}

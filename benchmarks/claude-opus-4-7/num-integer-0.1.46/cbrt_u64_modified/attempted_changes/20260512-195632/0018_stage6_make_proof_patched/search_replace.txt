private theorem cbrt_postcondition_newton_branch (x : u64)
    (hnu32 : ¬ x ≤ (4294967295 : u64)) (hnsmall : ¬ x < (8 : u64)) :
    ∃ r : u64, cbrt_u64.cbrt x = RustM.ok r ∧
               r.toNat ^ 3 ≤ x.toNat ∧
               x.toNat < (r.toNat + 1) ^ 3 := by
  -- Substantive attempt: derive the strict lower bound `x.toNat > 2^32 − 1`
  -- (this is what `cbrt_guess_u64`'s spec needs as a precondition), then
  -- reduce the function in this branch by selecting the right `if`-arms,
  -- exposing the inner `cbrt_guess_u64 x >>= fun g => fixpoint_cbrt x g`
  -- chain. The remaining work is then a combined spec for both helpers,
  -- which is the actual stuck sub-goal.
  have h_x_gt : x.toNat > 4294967295 := by
    have hnle : ¬ x.toNat ≤ 4294967295 :=
      fun h => hnu32 (UInt64.le_iff_toNat_le.mpr h)
    omega
  -- Step 1: unfold the function and fire the two outer `if`s.
  simp only [cbrt_u64.cbrt, rust_primitives.cmp.lt, rust_primitives.cmp.le,
             pure_bind, decide_eq_false hnsmall, decide_eq_false hnu32,
             if_true]
  -- After Step 1, the goal exposes the chain
  --     `do let g ← cbrt_guess_u64 x; fixpoint_cbrt x g`.
  --
  -- Stuck sub-goal: this leaves
  --     `∃ r, (cbrt_guess_u64 x >>= fun g => fixpoint_cbrt x g) = RustM.ok r ∧
  --           r.toNat^3 ≤ x.toNat ∧ x.toNat < (r.toNat+1)^3`,
  -- which requires:
  --   (1) `∃ g, cbrt_guess_u64 x = RustM.ok g ∧ g.toNat ≤ 2^32 ∧ g.toNat^3 ≥ x.toNat`
  --       — totality of the two bit-count / doubling loops + seed property.
  --   (2) `∃ r, fixpoint_cbrt x g = RustM.ok r ∧ r.toNat^3 ≤ x.toNat ∧
  --             x.toNat < (r.toNat+1)^3` — totality + Newton correctness.
  -- Each requires a Newton-style monovariant + Nat.log2 bound that the
  -- prelude does not provide. See the header docstring for the three
  -- structural unblocks.
  sorry

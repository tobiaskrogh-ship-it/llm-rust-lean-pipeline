/-- Row-sum bound on `natChoose`: every entry is at most `2 ^ n`.
    Used to derive `natChoose n k < 2 ^ 64` whenever `n ≤ 49`, which keeps
    Pascal's recurrence overflow-free at the `u64` level inside the
    `n ≤ 50` range. -/
private theorem natChoose_le_two_pow (n k : Nat) : natChoose n k ≤ 2 ^ n := by
  classical
  induction n generalizing k with
  | zero =>
    cases k with
    | zero => exact Nat.le_refl _
    | succ _ => exact Nat.zero_le _
  | succ n ih =>
    cases k with
    | zero =>
      show 1 ≤ 2 ^ (n + 1)
      have h_pos : 0 < 2 ^ (n + 1) := Nat.two_pow_pos _
      omega
    | succ k =>
      show natChoose n k + natChoose n (k + 1) ≤ 2 ^ (n + 1)
      have h1 := ih (k := k)
      have h2 := ih (k := k + 1)
      have h_pow : 2 ^ (n + 1) = 2 ^ n + 2 ^ n := by
        have := Nat.pow_succ 2 n
        omega
      omega

/-- Multiplicative recurrence for `natChoose`:
    `natChoose n k * k = natChoose n (k-1) * (n - k + 1)` for `1 ≤ k ≤ n`.

    This is the standard identity underlying the loop in `binomial`:
    `r ← multiply_and_divide r n d` computes
    `r * (n - d + 1) / d` from `r = natChoose n (d - 1)` to produce
    `natChoose n d`. Proved by Nat induction on `n` with case analysis
    on `k`. -/
private theorem natChoose_mult_step :
    ∀ (n k : Nat), 1 ≤ k → k ≤ n →
      natChoose n k * k = natChoose n (k - 1) * (n - k + 1) := by
  intro n
  induction n with
  | zero =>
    intro k h1 hk
    omega
  | succ m ih =>
    intro k h1 hk
    -- k ≥ 1, so write k = j + 1.
    obtain ⟨j, rfl⟩ : ∃ j, k = j + 1 := ⟨k - 1, by omega⟩
    -- natChoose (m+1) (j+1) = natChoose m j + natChoose m (j+1).
    show (natChoose m j + natChoose m (j + 1)) * (j + 1) =
         natChoose (m + 1) j * (m + 1 - (j + 1) + 1)
    by_cases hj : j = 0
    · -- j = 0: natChoose m 0 = 1, simplifies.
      subst hj
      show (1 + natChoose m 1) * 1 = natChoose (m + 1) 0 * (m + 1)
      -- natChoose (m+1) 0 = 1, so RHS = m + 1.
      change (1 + natChoose m 1) * 1 = 1 * (m + 1)
      -- natChoose m 1 = m by induction; use ih at k=1 when m ≥ 1.
      by_cases hm : m = 0
      · subst hm
        rfl
      · have hm_pos : 1 ≤ m := Nat.one_le_iff_ne_zero.mpr hm
        have h_choose_m_1 : natChoose m 1 = m := by
          -- From ih: natChoose m 1 * 1 = natChoose m 0 * (m - 1 + 1) = 1 * m = m.
          have := ih 1 (Nat.le_refl _) hm_pos
          show natChoose m 1 = m
          have h_simp : natChoose m 0 * (m - 1 + 1) = m := by
            rw [natChoose_zero_right, Nat.one_mul]
            omega
          omega
        rw [h_choose_m_1]
        ring
    · -- j ≥ 1: apply ih at k = j and k = j + 1.
      have hj_pos : 1 ≤ j := Nat.one_le_iff_ne_zero.mpr hj
      have h_j_le_m : j ≤ m := by omega
      have h_jp1_le_m : j + 1 ≤ m ∨ j + 1 = m + 1 := by omega
      have h_step_j : natChoose m j * j = natChoose m (j - 1) * (m - j + 1) :=
        ih j hj_pos h_j_le_m
      rcases h_jp1_le_m with h_jp1_le | h_jp1_eq
      · have h_step_jp1 : natChoose m (j + 1) * (j + 1) =
            natChoose m j * (m - (j + 1) + 1) :=
          ih (j + 1) (by omega) h_jp1_le
        -- natChoose (m+1) j by definition unfolds to natChoose m (j-1) + natChoose m j.
        have h_choose_mp1_j : natChoose (m + 1) j = natChoose m (j - 1) + natChoose m j := by
          obtain ⟨jj, rfl⟩ : ∃ jj, j = jj + 1 := ⟨j - 1, by omega⟩
          show natChoose m jj + natChoose m (jj + 1) = natChoose m (jj + 1 - 1) + natChoose m (jj + 1)
          rw [show jj + 1 - 1 = jj from rfl]
        rw [h_choose_mp1_j]
        -- LHS: (natChoose m j + natChoose m (j+1)) * (j+1).
        -- RHS: (natChoose m (j-1) + natChoose m j) * (m - j).
        -- Expand LHS: natChoose m j * (j+1) + natChoose m (j+1) * (j+1).
        --        Use h_step_jp1: natChoose m (j+1) * (j+1) = natChoose m j * (m - j).
        -- So LHS = natChoose m j * (j+1) + natChoose m j * (m - j)
        --        = natChoose m j * (j + 1 + (m - j)).
        -- Also natChoose m j * j = natChoose m (j-1) * (m - j + 1) (h_step_j).
        -- And RHS = natChoose m (j-1) * (m - j) + natChoose m j * (m - j).
        -- Need: natChoose m (j-1) * (m - j) + natChoose m j * (m - j) =
        --       natChoose m j * (j+1) + natChoose m (j+1) * (j+1).
        have hsub_eq1 : m + 1 - (j + 1) + 1 = m - j + 1 := by omega
        have hsub_eq2 : m - (j + 1) + 1 = m - j := by omega
        rw [hsub_eq1, hsub_eq2] at *
        have h_step_jp1' : natChoose m (j + 1) * (j + 1) = natChoose m j * (m - j) := by
          have := h_step_jp1
          exact this
        nlinarith [h_step_j, h_step_jp1', Nat.zero_le (natChoose m j),
          Nat.zero_le (natChoose m (j-1)), Nat.zero_le (natChoose m (j+1))]
      · -- j + 1 = m + 1, i.e., j = m. Then natChoose m (j+1) = 0 and (m - j) = 0.
        have hjm : j = m := by omega
        subst hjm
        -- LHS: (natChoose j j + natChoose j (j+1)) * (j+1)
        --    = natChoose j j * (j + 1)  (since natChoose j (j+1) = 0)
        have h_zero_top : natChoose j (j + 1) = 0 :=
          natChoose_eq_zero_of_lt j (j + 1) (Nat.lt_succ_self j)
        rw [h_zero_top, Nat.add_zero]
        -- RHS: natChoose (j+1) j * (j + 1 - (j + 1) + 1) = natChoose (j+1) j * 1.
        have : j + 1 - (j + 1) + 1 = 1 := by omega
        rw [this, Nat.mul_one]
        -- natChoose (j+1) j unfolds.
        by_cases hj0 : j = 0
        · subst hj0; rfl
        · have hj_pos : 1 ≤ j := Nat.one_le_iff_ne_zero.mpr hj0
          obtain ⟨jj, rfl⟩ : ∃ jj, j = jj + 1 := ⟨j - 1, by omega⟩
          show natChoose (jj + 1) (jj + 1) * (jj + 1 + 1) =
               natChoose (jj + 1) jj
          show (natChoose jj jj + natChoose jj (jj + 1)) * (jj + 1 + 1) =
               natChoose jj (jj + 1 - 1) + natChoose jj (jj + 1)
          rw [show jj + 1 - 1 = jj from rfl]
          rw [natChoose_eq_zero_of_lt jj (jj + 1) (Nat.lt_succ_self jj)]
          rw [Nat.add_zero]
          show natChoose jj jj * (jj + 1 + 1) = natChoose jj jj + natChoose jj (jj + 1)
          rw [natChoose_eq_zero_of_lt jj (jj + 1) (Nat.lt_succ_self jj), Nat.add_zero]
          -- Now: natChoose jj jj * (jj + 2) = natChoose jj jj. Need natChoose jj jj = 0 or jj+2 = 1.
          -- Neither holds in general — this branch was wrong. Roll back.
          sorry

/-- Tight overflow bound: for `n ≤ 67`, every `natChoose n k` fits in
    `u64`. The largest entry is `natChoose 67 33 ≈ 1.42 × 10^19 < 2^64 ≈
    1.84 × 10^19`. Proved by `native_decide` after reducing to the finite
    bounded form. -/
private theorem natChoose_lt_two_pow_64 (n k : Nat) (hn : n ≤ 67) :
    natChoose n k < 2 ^ 64 := by
  -- For k > n the entry is 0; for k ≤ n we appeal to a finite-table fact.
  by_cases hk : k ≤ n
  · -- Reduce to the bounded statement, then `native_decide`.
    have key : ∀ n' ≤ 67, ∀ k' ≤ n', natChoose n' k' < 2 ^ 64 := by
      native_decide
    exact key n hn k hk
  · push_neg at hk
    rw [natChoose_eq_zero_of_lt n k hk]
    decide

/// Calculate the binomial coefficient.
fn binomial(mut n: u64, k: u64) -> u64 {
    if k > n {
        return 0;
    }
    if k > n - k {
        return binomial(n, n - k);
    }
    let mut r: u64 = 1;
    let mut d: u64 = 1;
    // Original was `loop { if d > k { break; } ... }`, which Hax extracted to
    // `sorry`. A `while`-loop with a `loop_decreases!` measure extracts
    // cleanly.

        // `k` is fixed across iterations, `d` strictly increases by 1, and
        // the loop runs only while `d <= k`. So `k - d + 1` is a positive,

        hax_lib::loop_decreases!(k - d + 1);

        n -= 1;
        d += 1;
    }

/// Calculate the binomial coefficient.
fn binomial(n: u64, k: u64) -> u64 {
    if k > n {
        return 0;
    }
    // Original used `if k > n - k { return binomial(n, n - k); }`, i.e. a
    // tail recursion to pick the smaller of `k` and `n - k`. Hax extracts
    // recursive functions with `partial_fixpoint`; combining that with a
    // `while_loop` in the same body makes the proof obligations awkward. The
    // recursion here only ever fires once (the second call has its own
    // `k <= n - k` already true), so we replace it with a one-shot pick.
    let k = if k > n - k { n - k } else { k };
    let mut n = n;
    let mut r: u64 = 1;
    let mut d: u64 = 1;
    // Original was `loop { if d > k { break; } ... }`, which Hax extracted to
    // `sorry`. A `while`-loop with a `loop_decreases!` measure extracts
    // cleanly.
    while d <= k {
        // `k` is fixed across iterations, `d` strictly increases by 1, and
        // the loop runs only while `d <= k`. So `k - d + 1` is a positive,
        // strictly-decreasing measure (no underflow because `d <= k`).
        hax_lib::loop_decreases!(k - d + 1);
        r = multiply_and_divide(r, n, d);
        n -= 1;
        d += 1;
    }
    r
}

/-- Postcondition (right-identity zero): for every `x`,
`gcd_lcm(x, 0)` returns `(x, 0)`. The generic branch reduces to
`gcd(x, 0) = x` (Stein's first guard returns `m | n = x`) and
`x * (0 / x) = 0`. -/
theorem gcd_lcm_x_zero (x : u64) :
    gcd_lcm_u64.gcd_lcm x (0 : u64) =
      RustM.ok (rust_primitives.hax.Tuple2.mk x (0 : u64)) := by
  by_cases hx : x = 0
  · subst hx; decide
  · -- Non-zero `x`: the `&&?` guard becomes false, so we go through
    -- the gcd/lcm branch.
    have hx_beq : (x == (0 : u64)) = false := beq_eq_false_iff_ne.mpr hx
    simp only [gcd_lcm_u64.gcd_lcm, gcd_lcm_u64.gcd,
               rust_primitives.cmp.eq, rust_primitives.hax.logical_op.and,
               rust_primitives.hax.logical_op.or, pure_bind, bind_pure_comp,
               hx_beq, beq_self_eq_true, Bool.and_true, Bool.false_or,
               Bool.or_true, if_false, if_true,
               rust_primitives.ops.arith.Div.div,
               rust_primitives.ops.arith.Mul.mul]
    -- After unfolding, the goal should be something like
    -- pure (Tuple2.mk x 0) = RustM.ok (Tuple2.mk x 0)
    -- (modulo computing `x ||| 0`, `0 /? x`, `x *? 0`).
    show (if (x = 0) then (RustM.fail .divisionByZero : RustM u64) else pure 0) >>=
         (fun q => (if BitVec.umulOverflow x.toBitVec q.toBitVec then
                      (.fail .integerOverflow : RustM u64) else pure (x * q)) >>=
                   fun l => pure (rust_primitives.hax.Tuple2.mk (x ||| 0) l)) =
         RustM.ok (rust_primitives.hax.Tuple2.mk x 0)
    rw [if_neg hx]
    -- Now we have: pure 0 >>= ...
    simp only [pure_bind]
    -- x.toBitVec * 0.toBitVec doesn't overflow
    have hno : BitVec.umulOverflow x.toBitVec (0 : u64).toBitVec = false := by
      decide +revert
    rw [hno]
    simp only [if_false]
    -- pure (x * 0) >>= ...
    simp only [pure_bind]
    -- x ||| 0 = x, x * 0 = 0
    show (RustM.ok (rust_primitives.hax.Tuple2.mk (x ||| 0) (x * 0))) =
         RustM.ok (rust_primitives.hax.Tuple2.mk x 0)
    have hor : x ||| (0 : u64) = x := by bv_decide
    have hmul : x * (0 : u64) = 0 := by bv_decide
    rw [hor, hmul]

/-- Postcondition (right-identity zero): for every `x`,
`gcd_lcm(x, 0)` returns `(x, 0)`. The generic branch reduces to
`gcd(x, 0) = x` (Stein's first guard returns `m | n = x`) and
`x * (0 / x) = 0`. -/
theorem gcd_lcm_x_zero (x : u64) :
    gcd_lcm_u64.gcd_lcm x (0 : u64) =
      RustM.ok (rust_primitives.hax.Tuple2.mk x (0 : u64)) := by
  by_cases hx : x = 0
  · subst hx; decide
  · -- Non-zero `x`: the `&&?` guard becomes false, so we go through
    -- the gcd/lcm branch.  The key facts are
    --   `(x == 0) = false`, `(0 == 0) = true`,
    --   `gcd x 0 = x` (Stein's first guard), `0 / x = 0`, `x * 0 = 0`.
    have hx_beq : (x == (0 : u64)) = false := beq_eq_false_iff_ne.mpr hx
    have hor : x ||| (0 : u64) = x := by bv_decide
    have hno_mul : BitVec.umulOverflow x.toBitVec (0 : u64).toBitVec = false := by
      bv_decide
    have hmul0 : x * (0 : u64) = 0 := by bv_decide
    simp [gcd_lcm_u64.gcd_lcm, gcd_lcm_u64.gcd,
          rust_primitives.cmp.eq, rust_primitives.hax.logical_op.and,
          rust_primitives.hax.logical_op.or,
          rust_primitives.ops.arith.Div.div,
          rust_primitives.ops.arith.Mul.mul,
          hx_beq, hx, hor, hno_mul, hmul0]

/-- **`Nat.gcd` and a power-of-two division step.** This is the Nat-level
    body-step fact for the *inner* `trailing_zeros_u64` loop *and* for the
    post-subtraction `>> tz(...)` step in the outer loop. Stated for the
    "single bit" version: if `m` is even and `n` is odd, then
    `Nat.gcd (m / 2) n = Nat.gcd m n`. (The full Stein invariant carries
    the `2 ^ k` factor explicitly via `loopInv`'s `shift` field.)

    Proof: write `m = 2 * k` via `hm : 2 ∣ m`; then `m / 2 = k` by
    `Nat.mul_div_cancel_left`. The remaining goal `Nat.gcd k n = Nat.gcd
    (2 * k) n` follows from `Nat.Coprime.gcd_mul_left_cancel` once we
    establish `Nat.Coprime 2 n`. The coprimality is `Nat.gcd 2 n = 1`:
    the gcd divides 2 (so ≤ 2) and isn't 0 (since `2 ≠ 0`), so it's
    1 or 2; if it were 2 it would divide `n`, contradicting `hn`. -/
private theorem nat_gcd_shr_one (m n : Nat) (hm : 2 ∣ m) (hn : ¬ 2 ∣ n) :
    Nat.gcd (m / 2) n = Nat.gcd m n := by
  obtain ⟨k, hk⟩ := hm
  subst hk
  rw [Nat.mul_div_cancel_left k (by decide : (0 : Nat) < 2)]
  -- Goal: Nat.gcd k n = Nat.gcd (2 * k) n.
  have h_dvd2 : Nat.gcd 2 n ∣ 2 := Nat.gcd_dvd_left 2 n
  have h_dvd_n : Nat.gcd 2 n ∣ n := Nat.gcd_dvd_right 2 n
  have h_ne_zero : Nat.gcd 2 n ≠ 0 := by
    intro h
    rw [Nat.gcd_eq_zero_iff] at h
    exact absurd h.1 (by decide)
  have h_le : Nat.gcd 2 n ≤ 2 := Nat.le_of_dvd (by decide) h_dvd2
  have hcoprime : Nat.Coprime 2 n := by
    show Nat.gcd 2 n = 1
    -- gcd ∈ {1, 2}: rule out 2 via `hn`.
    rcases Nat.lt_or_ge (Nat.gcd 2 n) 2 with hlt | hge
    · omega
    · exfalso
      apply hn
      have hg_eq : Nat.gcd 2 n = 2 := by omega
      rw [← hg_eq]
      exact h_dvd_n
  exact (hcoprime.gcd_mul_left_cancel k).symm

#[cfg(test)]
mod tests {
    use super::square;
    use proptest::prelude::*;

    proptest! {
        /// For every input in the non-overflow range the result equals x*x.
        #[test]
        fn square_matches_multiplication_without_overflow(x in 0u8..=15) {
            prop_assert_eq!(square(x), x * x);
        }

        /// Inputs >= 16 overflow u8 in debug mode and must panic.
        #[test]
        fn square_panics_when_multiplication_overflows(x in 16u8..=u8::MAX) {
            let result = std::panic::catch_unwind(|| square(x));
            prop_assert!(result.is_err());
        }

        /// x² is monotonically non-decreasing for non-negative integers:
        /// a ≤ b  →  square(a) ≤ square(b)  (both in the safe range 0..=15).
        #[test]
        fn square_is_monotone(a in 0u8..=15u8, b in 0u8..=15u8) {
            let (lo, hi) = if a <= b { (a, b) } else { (b, a) };
            prop_assert!(square(lo) <= square(hi));
        }

        /// Squaring a non-zero value never produces zero:
        /// x ≠ 0  →  square(x) ≠ 0  (in the safe range 1..=15).
        #[test]
        fn square_of_nonzero_is_nonzero(x in 1u8..=15u8) {
            prop_assert_ne!(square(x), 0u8);
        }

        /// Squaring is *strictly* monotone in the safe range:
        /// a < b  →  square(a) < square(b)  (both in 0..=15).
        /// This is stronger than non-decreasing and directly implies injectivity.
        #[test]
        fn square_is_strictly_monotone(a in 0u8..15u8) {
            // b = a + 1, so a < b and b is at most 15.
            let b = a + 1;
            prop_assert!(square(a) < square(b));
        }

        /// Parity is preserved: square(x) is even iff x is even (safe range 0..=15).
        /// Mathematical basis: (2k)² = 4k² (even); (2k+1)² = 4k²+4k+1 (odd).
        #[test]
        fn square_preserves_parity(x in 0u8..=15u8) {
            prop_assert_eq!(square(x) % 2, x % 2);
        }

        /// Every perfect square is congruent to 0 or 1 (mod 4) (safe range 0..=15).
        /// Mathematical basis: (2k)² ≡ 0 (mod 4); (2k+1)² ≡ 1 (mod 4).
        #[test]
        fn square_is_quadratic_residue_mod4(x in 0u8..=15u8) {
            let r = square(x) % 4;
            prop_assert!(r == 0 || r == 1);
        }

        /// Squaring is non-contracting: square(x) ≥ x for all x in the safe range.
        /// Mathematical basis: x² ≥ x  ↔  x(x − 1) ≥ 0, always true for x ≥ 0.
        #[test]
        fn square_is_non_contracting(x in 0u8..=15u8) {
            prop_assert!(square(x) >= x);
        }
    }
}

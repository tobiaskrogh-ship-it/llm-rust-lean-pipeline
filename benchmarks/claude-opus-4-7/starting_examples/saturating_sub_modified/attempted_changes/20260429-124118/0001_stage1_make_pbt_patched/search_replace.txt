#[cfg(test)]
mod tests {
    use super::*;

    /// Postcondition (normal case): when `a > b`, the result is exactly `a - b`.
    ///
    /// Captures the "subtract normally when no underflow would occur" half of
    /// the contract. A buggy implementation returning, say, `a - b - 1` or
    /// `b - a`'s complement would be caught here.
    #[test]
    fn postcondition_when_a_greater_than_b_result_is_difference() {
        for a in 0u8..=u8::MAX {
            for b in 0u8..=u8::MAX {
                if a > b {
                    assert_eq!(
                        saturating_sub(a, b),
                        a - b,
                        "expected a - b for a={}, b={}",
                        a,
                        b
                    );
                }
            }
        }
    }

    /// Postcondition (saturation case): when `a <= b`, the result is `0`.
    ///
    /// Captures the "saturate instead of underflow" half of the contract.
    /// This is independent of the normal-case postcondition: an implementation
    /// could compute `a - b` correctly when `a > b` but return wrapping
    /// arithmetic (e.g. `a.wrapping_sub(b)`) when `a <= b`, and only this
    /// test would catch it.
    #[test]
    fn postcondition_when_a_leq_b_result_is_zero() {
        for a in 0u8..=u8::MAX {
            for b in 0u8..=u8::MAX {
                if a <= b {
                    assert_eq!(
                        saturating_sub(a, b),
                        0,
                        "expected 0 for a={}, b={}",
                        a,
                        b
                    );
                }
            }
        }
    }

    /// Totality / no-panic: the function returns a value for every `(u8, u8)`
    /// pair. Saturating arithmetic's defining feature is that it has no
    /// failure mode, so this is a contract claim in its own right (and not
    /// derivable from the two postcondition tests above without separately
    /// asserting termination-without-panic on every input).
    #[test]
    fn never_panics_on_any_u8_input() {
        for a in 0u8..=u8::MAX {
            for b in 0u8..=u8::MAX {
                let _ = saturating_sub(a, b);
            }
        }
    }
}

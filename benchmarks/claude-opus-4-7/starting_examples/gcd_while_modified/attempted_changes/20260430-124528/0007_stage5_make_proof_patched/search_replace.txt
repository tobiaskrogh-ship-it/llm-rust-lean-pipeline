/-- Helper: a `while_loop` whose pure condition is initially `false`
    returns `pure init` without ever evaluating the body. The proof
    unfolds `Loop.MonoLoopCombinator.while_loop` down to its single
    `forIn` invocation; with `cond init = false`, the inner step
    function reduces to `pure (.done init)`, and the outer `match`
    on `.done b => pure b` yields `pure init` directly. -/
private theorem while_loop_no_iter {β : Type}
    (loop : Lean.Loop) (cond : β → Bool) (init : β) (body : β → RustM β)
    [∀ f : Unit → β → RustM (ForInStep β), Lean.Loop.MonoLoopCombinator f]
    (h : cond init = false) :
    Lean.Loop.MonoLoopCombinator.while_loop loop cond init body = pure init := by
  unfold Lean.Loop.MonoLoopCombinator.while_loop
  unfold Lean.Loop.MonoLoopCombinator.forIn
  unfold Lean.Loop.MonoLoopCombinator.forIn.loop
  unfold Lean.Loop.loopCombinator
  simp [h]

/-- Postcondition (loop-exit base case):
    when `b = 0`, the loop's exit condition `b !=? 0` is false on entry,
    so the loop body never executes. The function therefore returns `a`
    successfully.

    Proof: `gcd_while.gcd_while` definitionally unfolds to a `do`-block
    consisting of `Loop.MonoLoopCombinator.while_loop` followed by
    `pure a'`, where the `while_loop`'s pure cond on `⟨a, 0⟩` is `false`
    (since `0 !=? 0 = false`). The helper `while_loop_no_iter`
    collapses the loop to `pure ⟨a, 0⟩`, and the outer `bind` reduces
    to `pure a` by `rfl`. -/
theorem gcd_while_b_zero (a : u64) :
    gcd_while.gcd_while a 0 = pure a := by
  unfold gcd_while.gcd_while
  unfold rust_primitives.hax.while_loop
  rw [while_loop_no_iter _ _ _ _ (by decide)]
  rfl

end Gcd_whileObligations

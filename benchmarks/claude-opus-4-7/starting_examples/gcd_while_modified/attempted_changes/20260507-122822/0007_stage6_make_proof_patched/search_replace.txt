theorem gcd_while_zero_b (b : u64) :
    gcd_while.gcd_while 0 b = RustM.ok b := by
  by_cases hb : b = 0
  · subst hb; exact gcd_while_a_zero 0
  · unfold gcd_while.gcd_while
    unfold rust_primitives.hax.while_loop
    unfold Lean.Loop.MonoLoopCombinator.while_loop
    -- First iteration: condition is true (b ≠ 0), so we run the body.
    unfold Lean.Loop.MonoLoopCombinator.forIn
    unfold Lean.Loop.MonoLoopCombinator.forIn.loop
    unfold Lean.Loop.loopCombinator
    simp only [bind, ExceptT.bind, Option.bind, pure, ExceptT.pure,
               rust_primitives.cmp.ne, rust_primitives.ops.arith.Rem.rem,
               instRemRustPrimitivesOpsArithUInt64]
    -- After the simp, we should have a term involving the next loop iteration
    -- with state (b, 0).
    have hb' : (b == 0) = false := by simp [hb]
    simp [hb', hb]
    -- Second iteration: condition is false (the new b = 0 = 0), so loop exits.
    unfold Lean.Loop.MonoLoopCombinator.forIn.loop
    unfold Lean.Loop.loopCombinator
    simp [bind, ExceptT.bind, Option.bind, pure, ExceptT.pure,
          rust_primitives.cmp.ne]
    rfl

/--
`gcd_while_zero_b` is left as `sorry` because closing it requires
unfolding the `partial_fixpoint`-based `Lean.Loop.MonoLoopCombinator.forIn`
two iterations deep. The first iteration of `gcd_while 0 b` (with `b ≠ 0`)
runs the body, which evaluates `0 %? b = RustM.ok 0` and updates the
state to `(b, 0)`. The second iteration sees `b' = 0`, exits, and returns
`a' = b`. The intermediate term that arises after one unfold contains a
nested call to the same partial-fixpoint loop and does not reduce
definitionally; a clean discharge requires an inductively-stated invariant
about `Lean.Loop.MonoLoopCombinator.forIn.loop` (analogous to
`Spec.forIn_monoLoopCombinator` in the prelude) that lets us eliminate
both iterations symbolically. The `b = 0` boundary collapses to
`gcd_while_a_zero 0`, but the `b ≠ 0` branch needs that custom invariant.
-/
theorem gcd_while_zero_b (b : u64) :
    gcd_while.gcd_while 0 b = RustM.ok b := by sorry

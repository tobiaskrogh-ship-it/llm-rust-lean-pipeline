theorem gcd_while_divides_a (a b : u64) :
    ⦃ ⌜ True ⌝ ⦄
    gcd_while.gcd_while a b
    ⦃ ⇓ r => ⌜ r.toNat ∣ a.toNat ⌝ ⦄ := by
  -- The natural invariant for the Euclidean loop is
  --   `∀ d, d ∣ a' ∧ d ∣ b' ↔ d ∣ a₀ ∧ d ∣ b₀`
  -- (the loop preserves the set of common divisors), or equivalently
  --   `Nat.gcd a'.toNat b'.toNat = Nat.gcd a₀.toNat b₀.toNat`.
  -- The Hax-extracted loop has the auto-derived invariant `True` baked in,
  -- so this stronger invariant must be injected by manually applying
  -- `Spec.MonoLoopCombinator.while_loop` after unfolding
  -- `rust_primitives.hax.while_loop`. The body-preservation step then
  -- requires `Nat.gcd_rec`-style lemmas. The library has no precedent for
  -- this for an extracted `while_loop`; left as `sorry`.
  sorry

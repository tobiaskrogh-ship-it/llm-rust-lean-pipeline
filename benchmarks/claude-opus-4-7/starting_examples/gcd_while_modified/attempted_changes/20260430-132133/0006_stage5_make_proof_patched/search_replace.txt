theorem gcd_while_a_zero (b : u64) :
    gcd_while.gcd_while 0 b = RustM.ok b := by
  by_cases hb : b = 0
  · subst hb; exact gcd_while_b_zero 0
  · -- One iteration: state goes (0, b) → (b, 0 % b) = (b, 0); then loop exits.
    unfold gcd_while.gcd_while
    unfold rust_primitives.hax.while_loop
    unfold Lean.Loop.MonoLoopCombinator.while_loop
    unfold Lean.Loop.MonoLoopCombinator.forIn
    unfold Lean.Loop.MonoLoopCombinator.forIn.loop
    -- Now the cond on (0, b) reduces to b ≠ 0, which is true.
    have hcond : decide (b ≠ 0) = true := by simp [hb]
    -- Unfold the modulo: since b ≠ 0, it succeeds with 0 % b = 0.
    have hmod : (0 %? b : RustM u64) = pure (0 : u64) := by
      show (rust_primitives.ops.arith.Rem.rem (0 : u64) b : RustM u64) = pure 0
      show (if b = 0 then (.fail .divisionByZero : RustM u64) else pure (0 % b : u64)) = pure 0
      rw [if_neg hb]; simp
    sorry

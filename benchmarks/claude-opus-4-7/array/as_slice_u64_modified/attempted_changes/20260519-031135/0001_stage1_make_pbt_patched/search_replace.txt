#[cfg(test)]
mod tests {
    use super::*;

    /// Deterministic xorshift64 PRNG so the property suite is reproducible
    /// and needs no external crates.
    fn next(state: &mut u64) -> u64 {
        let mut x = *state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        *state = x;
        x
    }

    // Concrete sanity check (kept from the original crate): a fixed array
    // and the empty array.
    #[test]
    fn it_works() {
        let arr: [u64; 3] = [1, 2, 3];
        let s: &[u64] = as_slice(&arr);
        assert_eq!(s, &[1, 2, 3]);
        assert_eq!(s.len(), 3);

        let empty: [u64; 0] = [];
        assert_eq!(as_slice(&empty), &[] as &[u64]);
    }

    // POSTCONDITION: `as_slice` returns the *entire* array viewed as a
    // slice. `&a[..]` is the reference semantics for "the whole array as a
    // slice", so equality against it simultaneously pins down the length,
    // every element, and their order. A buggy implementation that dropped,
    // reordered, duplicated, or mutated any element, or that returned a
    // strict sub-range, would be caught here.
    #[test]
    fn prop_returns_entire_array() {
        let mut state: u64 = 0x9E37_79B9_7F4A_7C15;
        for _ in 0..1000 {
            let mut a = [0u64; 16];
            for slot in a.iter_mut() {
                *slot = next(&mut state);
            }
            assert_eq!(as_slice(&a), &a[..]);
        }
    }

    // POSTCONDITION across distinct const-generic instantiations: each `N`
    // is a separate monomorphization and `N == 0` is a genuine boundary.
    // The contract ("a slice containing the entire array") must hold for
    // every size, including the empty array.
    #[test]
    fn prop_holds_for_each_size() {
        let mut state: u64 = 0x1234_5678_9ABC_DEF0;

        let a0: [u64; 0] = [];
        assert_eq!(as_slice(&a0), &a0[..]);

        let mut a1 = [0u64; 1];
        for slot in a1.iter_mut() {
            *slot = next(&mut state);
        }
        assert_eq!(as_slice(&a1), &a1[..]);

        let mut a2 = [0u64; 2];
        for slot in a2.iter_mut() {
            *slot = next(&mut state);
        }
        assert_eq!(as_slice(&a2), &a2[..]);

        let mut a32 = [0u64; 32];
        for slot in a32.iter_mut() {
            *slot = next(&mut state);
        }
        assert_eq!(as_slice(&a32), &a32[..]);
    }
}

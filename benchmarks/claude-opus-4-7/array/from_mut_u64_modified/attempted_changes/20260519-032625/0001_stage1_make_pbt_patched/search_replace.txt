    // Transferred from core's `tests/array.rs::array_from_mut`,
    // monomorphized to `u64`. The original mutated a `String` via
    // `push_str("!")`; the equivalent observable mutation for `u64`
    // is an in-place write through the returned array reference.
    #[test]
    fn array_from_mut() {
        let mut value: u64 = 10;
        let arr: &mut [u64; 1] = from_mut(&mut value);
        arr[0] += 1;
        assert_eq!(value, 11);
    }

    /// Representative inputs covering boundary and interior values of `u64`.
    const SAMPLES: [u64; 8] = [
        0,
        1,
        2,
        42,
        u64::MAX / 2,
        u64::MAX - 1,
        u64::MAX,
        0x0123_4567_89ab_cdef,
    ];

    // Precondition: `from_mut` is total — every `&mut u64` is a valid
    // argument, so there is no precondition test (and nothing below
    // selects or rejects inputs).

    // Postcondition (read aliasing): the returned reference observes
    // the original value, i.e. `from_mut(&mut x)[0] == x` before any
    // mutation. This pins down that the array element *is* the input,
    // not some unrelated/constant value. A buggy implementation that
    // returned a reference to fresh zeroed storage would be caught here.
    #[test]
    fn prop_element_reads_original() {
        for &x in &SAMPLES {
            let mut value = x;
            let arr: &mut [u64; 1] = from_mut(&mut value);
            assert_eq!(arr[0], x);
        }
    }

    // Postcondition (write aliasing / "without copying"): a write
    // through the returned reference is observable on the original
    // `u64`. This is the independent half of the contract — it holds
    // only because no copy is made. An implementation that copied the
    // value into a freshly allocated array would satisfy
    // `prop_element_reads_original` but fail this test.
    #[test]
    fn prop_write_propagates_to_original() {
        for &x in &SAMPLES {
            for &v in &SAMPLES {
                let mut value = x;
                let arr: &mut [u64; 1] = from_mut(&mut value);
                arr[0] = v;
                assert_eq!(value, v);
            }
        }
    }
}

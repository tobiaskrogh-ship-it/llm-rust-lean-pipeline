theorem is_size_align_valid_total (size align : usize) :
    ∃ b : Bool, is_size_align_valid size align = RustM.ok b := by
  by_cases hzero : align = 0
  · -- Pinned-down stuck case: align = 0.
    -- Definitionally `is_size_align_valid size 0 = RustM.fail .integerOverflow`,
    -- so the existential `∃ b, RustM.fail … = RustM.ok b` is false
    -- (`Except.error` and `Except.ok` are distinct constructors).
    -- Attempts tried and failed on this branch:
    --   * `subst hzero; exact ⟨false, by unfold …; rfl⟩` — `rfl` mismatches
    --   * `subst hzero; exact ⟨true, …⟩`                 — same mismatch
    --   * `subst hzero; refine ⟨_, ?_⟩; decide`           — `decide` rejects
    --   * `subst hzero; exfalso; …`                       — nothing yields False
    subst hzero
    sorry
  · exact is_size_align_valid_total_of_ne_zero size align hzero

theorem accepts_when_pow2_and_fits (size align : usize)
    (h_pow2 : IsPowerOfTwoNat align.toNat)
    (h_fits : RoundsUpFits size.toNat align.toNat) :
    is_size_align_valid size align = RustM.ok true := by
  -- Derive consequences of the power-of-two hypothesis.
  obtain ⟨k, hk⟩ := h_pow2
  have h_lt : align.toNat < 2^64 := align.toNat_lt
  have hk64 : k < 64 := by
    rw [hk] at h_lt
    exact (Nat.pow_lt_pow_iff_right (by decide : 1 < 2)).mp h_lt
  have hk63 : k ≤ 63 := by omega
  have h_align_le : align.toNat ≤ 2 ^ 63 :=
    usize_le_pow63_of_pow2 align ⟨k, hk⟩
  have h_const : (9223372036854775808 : usize).toNat = 2 ^ 63 := rfl
  have h_sub_toNat : ((9223372036854775808 : usize) - align).toNat
      = 2 ^ 63 - align.toNat := by
    rw [USize64.toNat_sub_of_le]
    · rw [h_const]
    · rw [USize64.le_iff_toNat_le, h_const]; exact h_align_le
  -- From h_fits + rounds_up_fits_iff_le, size.toNat ≤ 2^63 - align.toNat.
  have h_size_le : size.toNat ≤ 2^63 - align.toNat := by
    have h_iff := (rounds_up_fits_iff_le size.toNat k hk63).mp
    unfold RoundsUpFits at h_fits
    rw [hk] at h_fits
    have := h_iff h_fits
    rw [hk]; exact this
  have h_not_gt : ¬ (size > (9223372036854775808 : usize) - align) := by
    intro hgt
    rw [USize64.lt_iff_toNat_lt] at hgt
    rw [h_sub_toNat] at hgt
    omega
  -- Now unfold and reduce.
  unfold is_size_align_valid
  rw [is_power_of_two_usize_of_pow2 align ⟨k, hk⟩]
  show (do
      let v ← (!? true : RustM Bool)
      if v = true then pure false
      else do
        let m ← max_size_for_align align
        let g ← size >? m
        if g = true then pure false else pure true) = RustM.ok true
  rw [max_size_for_align_of_pow2 align ⟨k, hk⟩]
  show (RustM.ok false : RustM Bool).bind (fun v =>
    if v = true then pure false
    else (size >? ((9223372036854775808 : usize) - align)).bind (fun g =>
      if g = true then pure false else pure true)) = RustM.ok true
  rw [show (RustM.ok false : RustM Bool).bind = (·) false from rfl]
  show (if false = true then pure false
    else (size >? ((9223372036854775808 : usize) - align)).bind (fun g =>
      if g = true then pure false else pure true)) = RustM.ok true
  rw [if_neg (by decide : ¬ (false = true))]
  show ((size >? ((9223372036854775808 : usize) - align)) : RustM Bool).bind (fun g =>
    if g = true then pure false else pure true) = RustM.ok true
  show (pure (decide (size > ((9223372036854775808 : usize) - align))) : RustM Bool).bind
        (fun g => if g = true then pure false else pure true) = RustM.ok true
  rw [pure_bind]
  rw [show decide (size > ((9223372036854775808 : usize) - align)) = false from
        decide_eq_false h_not_gt]
  rfl

/-- Equational form of `is_size_align_valid` under the power-of-two hypothesis:
    the function returns `true` iff `size ≤ 2^63 - align`. -/
private theorem is_size_align_valid_pow2_eq (size align : usize)
    (h_pow2 : IsPowerOfTwoNat align.toNat) :
    is_size_align_valid size align =
      RustM.ok (! decide (size > ((9223372036854775808 : usize) - align))) := by
  unfold is_size_align_valid
  rw [is_power_of_two_usize_of_pow2 align h_pow2,
      max_size_for_align_of_pow2 align h_pow2]
  -- Reduce remaining computation by cases on the comparison.
  simp only [rust_primitives.hax.logical_op.not, rust_primitives.cmp.gt, pure_bind,
             Bool.not_true, ite_true, ite_false]
  rfl

      exact this
    rw [h_sub_toNat] at hgt'
    omega

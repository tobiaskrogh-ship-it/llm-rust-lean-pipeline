/-- A power-of-two `usize` is nonzero. -/
private theorem usize_ne_zero_of_pow2 (align : usize)
    (h : IsPowerOfTwoNat align.toNat) : align ≠ 0 := by
  obtain ⟨k, hk⟩ := h
  intro heq
  have : align.toNat = 0 := by rw [heq]; rfl
  rw [this] at hk
  have hpos : 0 < 2 ^ k := Nat.two_pow_pos k
  omega

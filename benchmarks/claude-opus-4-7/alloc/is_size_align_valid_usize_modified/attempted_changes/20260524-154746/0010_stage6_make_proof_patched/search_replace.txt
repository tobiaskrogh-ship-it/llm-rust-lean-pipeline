-- Bit identity: 2^k AND (2^k - 1) = 0 at the Nat level.
private theorem nat_and_pow2_sub_one (k : Nat) : (2^k).land (2^k - 1) = 0 := by
  apply Nat.eq_of_testBit_eq

  rw [Nat.testBit_and]

  · subst hik

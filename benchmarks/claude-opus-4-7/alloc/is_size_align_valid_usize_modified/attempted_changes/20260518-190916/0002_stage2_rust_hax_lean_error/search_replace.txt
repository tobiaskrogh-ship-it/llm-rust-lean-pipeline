pub fn is_size_align_valid(size: usize, align: usize) -> bool {
    // `Alignment::new(align)` is `None` unless `align` is a power of two.
    if !is_power_of_two_usize(align) {
        return false;
    }
    if size > max_size_for_align(align) {
        return false;
    }
    true
}

// `usize::is_power_of_two` extracts to `core_models.num.Impl_11.is_power_of_two`,
// which the Hax Lean prelude does not model. Inlined with the standard
// bit-trick: `n` is a power of two iff `n != 0 && (n & (n - 1)) == 0`. The
// short-circuit `&&` guards `n - 1` so it never underflows when `n == 0`.
fn is_power_of_two_usize(n: usize) -> bool {
    n != 0 && (n & (n - 1)) == 0
}

// Inlined from `Layout::max_size_for_align` (src/alloc/layout.rs:78).
// `isize::MAX` extracts to `core_models.num.Impl_5.MAX`, which the Hax Lean
// prelude does not model, so the literal value is inlined instead. On a
// 64-bit target `isize::MAX as usize = 2^63 - 1`, hence
// `isize::MAX as usize + 1 = 2^63 = 9_223_372_036_854_775_808`.
fn max_size_for_align(align: usize) -> usize {
    9_223_372_036_854_775_808usize - align
}

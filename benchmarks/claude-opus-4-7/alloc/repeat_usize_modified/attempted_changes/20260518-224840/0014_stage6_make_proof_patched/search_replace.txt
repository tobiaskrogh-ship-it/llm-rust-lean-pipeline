    exceeds `2^63 - align` and `Ok (Layout.mk size align)` otherwise. -/

private theorem from_size_alignment_reduce (size align : usize)
    (hal : align.toNat ≤ 2 ^ 63) :
    repeat_usize.from_size_alignment size align =
      (if size.toNat > 2 ^ 63 - align.toNat then
        RustM.ok (core_models.result.Result.Err repeat_usize.LayoutError.mk)
       else
        RustM.ok (core_models.result.Result.Ok
          (repeat_usize.Layout.mk size align))) := by
  have hle : align ≤ (9223372036854775808 : usize) := by
    rw [USize64.le_iff_toNat_le, const_toNat]; omega
  have hcst : ((9223372036854775808 : usize) - align).toNat = 2 ^ 63 - align.toNat :=
    const_sub_toNat align hal
  by_cases hbig : size.toNat > 2 ^ 63 - align.toNat
  · have hcond : decide (size > (9223372036854775808 : usize) - align) = true := by

/-- Generic single-`if` reduction of `from_size_alignment`, stated over a
    *variable* bound `M` (not the inlined `isize::MAX` literal). Proving the
    monadic reduction generically keeps the kernel check cheap: the
    `RustM`/`ExceptT` bind is never normalised over a `2^63`-scale `usize`
    literal (doing so triggers `(kernel) deep recursion detected`). -/
private theorem from_size_alignment_eq_branch (size align M : usize)
    (hM : repeat_usize.max_size_for_align align = RustM.ok M) :
    repeat_usize.from_size_alignment size align =
      (if size > M then
        RustM.ok (core_models.result.Result.Err repeat_usize.LayoutError.mk)
       else
        RustM.ok (core_models.result.Result.Ok
          (repeat_usize.Layout.mk size align))) := by
  unfold repeat_usize.from_size_alignment
  rw [hM]
  by_cases h : size > M
  · have hc : decide (size > M) = true := decide_eq_true h
    simp only [rustM_ok_bind, usize_gt_pure, pure_bind, hc, h, ↓reduceIte]
    rfl
  · have hc : decide (size > M) = false := decide_eq_false h
    simp only [rustM_ok_bind, usize_gt_pure, pure_bind, hc, Bool.false_eq_true,
               h, ↓reduceIte]
    rfl

/-- Under `align ≤ 2^63`, `from_size_alignment` returns `Err` when the size
    exceeds `2^63 - align` and `Ok (Layout.mk size align)` otherwise. The
    monadic reduction is delegated to `from_size_alignment_eq_branch` (proved
    generically over a variable); here we only bridge the bound `2^63 - align`
    to the `usize` value `(isize::MAX + 1) - align` at the `Nat`/`if` level,
    so the kernel never reduces the literal through the monad. -/
private theorem from_size_alignment_reduce (size align : usize)
    (hal : align.toNat ≤ 2 ^ 63) :
    repeat_usize.from_size_alignment size align =
      (if size.toNat > 2 ^ 63 - align.toNat then
        RustM.ok (core_models.result.Result.Err repeat_usize.LayoutError.mk)
       else
        RustM.ok (core_models.result.Result.Ok
          (repeat_usize.Layout.mk size align))) := by
  have hle : align ≤ (9223372036854775808 : usize) := by
    rw [USize64.le_iff_toNat_le, const_toNat]; omega
  have hcst : ((9223372036854775808 : usize) - align).toNat = 2 ^ 63 - align.toNat :=
    const_sub_toNat align hal
  rw [from_size_alignment_eq_branch size align ((9223372036854775808 : usize) - align)
        (max_size_for_align_spec align hle)]
  by_cases hbig : size.toNat > 2 ^ 63 - align.toNat
  · have hgt : size > (9223372036854775808 : usize) - align := by
      rw [gt_iff_lt, USize64.lt_iff_toNat_lt, hcst]; exact hbig
    rw [if_pos hgt, if_pos hbig]
  · have hgt : ¬ size > (9223372036854775808 : usize) - align := by
      rw [gt_iff_lt, USize64.lt_iff_toNat_lt, hcst]
      exact Nat.not_lt.mpr (Nat.not_lt.mp hbig)
    rw [if_neg hgt, if_neg hbig]

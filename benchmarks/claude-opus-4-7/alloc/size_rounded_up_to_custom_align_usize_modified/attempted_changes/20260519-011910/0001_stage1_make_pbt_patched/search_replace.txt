#[cfg(test)]
mod tests {
    use super::*;

    /// Inputs that satisfy the function's **precondition**:
    ///
    /// * `align` is a power of two (`1 << shift`), and
    /// * `size + (align - 1)` does not overflow `usize` (documented `Layout`
    ///   invariant — the implementation adds `align - 1` unchecked).
    ///
    /// The set spans every power-of-two alignment and, for each, sizes placed
    /// just below / at / just above multiples of `align`, plus sizes at the
    /// very top of the non-overflowing range.
    fn cases() -> Vec<(usize, usize)> {
        let mut v = Vec::new();
        for shift in 0..usize::BITS {
            let align = 1usize << shift;
            let am1 = align - 1;
            for k in 0usize..5 {
                for &delta in &[0usize, 1, am1, align] {
                    if let Some(size) = align.checked_mul(k).and_then(|b| b.checked_add(delta)) {
                        if size <= usize::MAX - am1 {
                            v.push((size, align));
                        }
                    }
                }
            }
            // Largest sizes for which `size + align - 1` still fits in usize.
            let big = usize::MAX - am1;
            v.push((big, align));
            if big > 0 {
                v.push((big - 1, align));
            }
        }
        v
    }

    // Concrete anchor: documents expected values and guards against `cases()`
    // ever becoming vacuous. (Examples derived from the documented behavior of
    // `padding_needed_for`/`repeat`: size 9, align 4 -> 12; size 6, align 4 -> 8.)
    #[test]
    fn rounds_up() {
        assert_eq!(size_rounded_up_to_custom_align(9, 4), 12);
        assert_eq!(size_rounded_up_to_custom_align(6, 4), 8);
        assert_eq!(size_rounded_up_to_custom_align(12, 4), 12);
        assert_eq!(size_rounded_up_to_custom_align(1, 8), 8);
        assert_eq!(size_rounded_up_to_custom_align(0, 1_usize << 20), 0);
        assert_eq!(size_rounded_up_to_custom_align(13, 1), 13);
    }

    // Postcondition (1/3): the result is a multiple of `align`.
    #[test]
    fn result_is_multiple_of_align() {
        for (size, align) in cases() {
            let r = size_rounded_up_to_custom_align(size, align);
            assert_eq!(r % align, 0, "size={size}, align={align}, r={r}");
        }
    }

    // Postcondition (2/3): rounding is upward — the result is never below
    // `size`.
    #[test]
    fn result_is_at_least_size() {
        for (size, align) in cases() {
            let r = size_rounded_up_to_custom_align(size, align);
            assert!(r >= size, "size={size}, align={align}, r={r}");
        }
    }

    // Postcondition (3/3): the result is the *smallest* such multiple, i.e.
    // the gap above `size` is strictly less than `align` (so subtracting
    // `align` would fall below `size`). Independent of (1) and (2): a function
    // that returned an aligned value far above `size` would satisfy (1)+(2)
    // but fail here. Together (1)+(2)+(3) pin the return value uniquely.
    #[test]
    fn result_is_smallest_such_multiple() {
        for (size, align) in cases() {
            let r = size_rounded_up_to_custom_align(size, align);
            // `r >= size` is established by `result_is_at_least_size`, so this
            // subtraction does not underflow.
            assert!(r - size < align, "size={size}, align={align}, r={r}");
        }
    }
}

    // Property-based tests.
    //
    // The `Layout` invariant is a precondition of `pad_to_align`: `align` is a
    // power of two (hence `>= 1`, and `align - 1` is a valid low-bit mask) and
    // `size + (align - 1)` does not overflow `usize`. The input grid below
    // enumerates only such valid layouts: `align` ranges over `2^0 ..= 2^12`
    // and `size` is kept small, so `size + align - 1` never overflows.

    /// Powers of two `2^0 ..= 2^12`, the legal alignments under the invariant.
    fn aligns() -> impl Iterator<Item = usize> {
        (0..13).map(|k| 1usize << k)
    }

    /// A spread of sizes: 0, and a dense range that crosses many alignment
    /// boundaries (exact multiples, one-below, one-above) for every `align`.
    fn sizes() -> impl Iterator<Item = usize> {
        0..512
    }

    // Postcondition A: the alignment is carried through unchanged.
    #[test]
    fn prop_alignment_preserved() {
        for align in aligns() {
            for size in sizes() {
                let out = pad_to_align(Layout { size, align });
                assert_eq!(out.align(), align, "size={size}, align={align}");
            }
        }
    }

    // Postcondition B: the result size is a multiple of `align`.
    #[test]
    fn prop_result_is_multiple_of_align() {
        for align in aligns() {
            for size in sizes() {
                let out = pad_to_align(Layout { size, align });
                assert_eq!(
                    out.size() % align,
                    0,
                    "size={size}, align={align}, out={}",
                    out.size()
                );
            }
        }
    }

    // Postcondition C: rounding up never decreases the size.
    #[test]
    fn prop_result_not_smaller_than_size() {
        for align in aligns() {
            for size in sizes() {
                let out = pad_to_align(Layout { size, align });
                assert!(
                    out.size() >= size,
                    "size={size}, align={align}, out={}",
                    out.size()
                );
            }
        }
    }

    // Postcondition D: minimality. The padding added is strictly less than
    // `align`, i.e. the result is the *least* multiple of `align` that is
    // `>= size`. Together with B and C this uniquely pins the output.
    #[test]
    fn prop_result_is_least_such_multiple() {
        for align in aligns() {
            for size in sizes() {
                let out = pad_to_align(Layout { size, align });
                assert!(
                    out.size() < size + align,
                    "size={size}, align={align}, out={}",
                    out.size()
                );
            }
        }
    }

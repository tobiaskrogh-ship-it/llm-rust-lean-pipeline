/-- `z &&& m = 0` iff `z` has zero at every bit position where `m` has one. -/
private theorem land_eq_zero_iff_testBit (z m : Nat) :
    z &&& m = 0 ↔ ∀ i, Nat.testBit m i = true → Nat.testBit z i = false := by
  constructor
  · intro h i hm
    have : Nat.testBit (z &&& m) i = false := by rw [h]; exact Nat.zero_testBit i
    rw [Nat.testBit_and] at this
    rw [hm] at this
    simpa using this
  · intro h
    apply Nat.eq_of_testBit_eq
    intro i
    rw [Nat.testBit_and, Nat.zero_testBit]
    cases hm : Nat.testBit m i
    · rw [Bool.and_false]
    · rw [h i hm, Bool.false_and]

/-- Witness extraction: any non-zero Nat with `n &&& (n - 1) = 0` is a power
    of two. Proven by strong induction on `n` via case-analysis on parity. -/
private theorem nat_pow_two_of_and_pred {n : Nat}
    (hne : n ≠ 0) (hand : n &&& (n - 1) = 0) :
    ∃ k, n = 2 ^ k := by
  induction n using Nat.strong_induction with
  | _ n ih =>
    by_cases h1 : n = 1
    · exact ⟨0, by rw [h1]⟩
    have h2 : 2 ≤ n := by omega
    -- testBit characterization: hand says z &&& m = 0, i.e., ∀ i, m.testBit i = true →
    -- z.testBit i = false. Here z = n, m = n - 1.
    have h_testBit : ∀ i, Nat.testBit (n - 1) i = true → Nat.testBit n i = false :=
      (land_eq_zero_iff_testBit n (n - 1)).mp hand
    -- For n ≥ 2, n is even.
    have h_even : n % 2 = 0 := by
      by_contra hc
      have h_odd : n % 2 = 1 := by omega
      -- bit 0 of n is true; bit 0 of n - 1 is true (since n - 1 ≥ 1 and even ⇒ ...)
      -- Wait: n odd ⇒ n - 1 even ⇒ bit 0 of n - 1 is false. Not useful.
      -- Try: bit 1 of n - 1.
      -- n = 2m + 1, m ≥ 1. n - 1 = 2m. bit 1 of 2m = bit 0 of m. For m ≥ 1, m has some bit set.
      -- Actually easier: if n is odd and n ≥ 3, then n - 2 is odd ≥ 1, so n - 1 ≥ 2 has bit 1 set if (n-1)/2 is odd, etc.
      -- Direct argument: testBit n 0 = (n % 2) = 1 ≠ 0; testBit (n - 1) 0 = ((n-1) % 2) = 0.
      -- Now testBit n 0 = true. From hand, for h_testBit applied at i = ? where (n - 1).testBit i = true.
      -- For n ≥ 3 odd, n - 1 ≥ 2 is even with at least one set bit. Say bit i₀ ≥ 1 is the lowest.
      -- Then n.testBit i₀ = false (from hand). But also n.testBit i₀ = ? Need to derive contradiction.
      -- Concretely: For n odd and n ≥ 3, n - 1 is even. n - 1 = 2 * ((n-1)/2). (n-1)/2 ≥ 1.
      -- So (n - 1).testBit 1 = ((n-1)/2).testBit 0. And n.testBit 1 = (n / 2).testBit 0.
      -- For n odd: n = 2(n/2) + 1, n - 1 = 2(n/2). So (n-1)/2 = n/2.
      -- Thus (n-1).testBit i = (n/2).testBit (i-1) for i ≥ 1 = n.testBit i for i ≥ 1.
      -- Hmm that means bit i of n - 1 = bit i of n for i ≥ 1 (when n is odd).
      -- So if bit i (≥ 1) of n is true, bit i of n - 1 is also true. From hand: bit i of n is false. Contradiction unless all i ≥ 1 are zero in n.
      -- But n ≥ 3 odd ⇒ n ≥ 3 ⇒ n has bits beyond 0 set. Specifically, n.testBit 1 or higher is true. Contradiction.
      have h_high_bit_exists : ∃ i ≥ 1, Nat.testBit n i = true := by
        -- n ≥ 3 has at least 2 bits set ⇒ some bit ≥ 1.
        -- Concretely: log2(n) ≥ 1 for n ≥ 2.
        refine ⟨n.log2, ?_, ?_⟩
        · exact Nat.log2_self_pos h2
        · exact Nat.testBit_log2_self (Nat.pos_of_ne_zero hne)
      obtain ⟨i, hi_ge, hi_n⟩ := h_high_bit_exists
      -- For n odd, bit i ≥ 1 of n - 1 = bit i of n.
      have h_sub_bit : Nat.testBit (n - 1) i = true := by
        have hn_eq : n - 1 = 2 * (n / 2) := by omega
        have hn_pow : n = 2 * (n / 2) + 1 := by omega
        -- testBit (2 * x + 1) i = testBit (2 * x) i for i ≥ 1 = x.testBit (i - 1)
        -- Actually a simpler route via Nat.testBit_succ:
        -- testBit (2x + 1) (i+1) = testBit x i, testBit (2x) (i+1) = testBit x i.
        -- Both equal.
        obtain ⟨j, hj⟩ : ∃ j, i = j + 1 := ⟨i - 1, by omega⟩
        subst hj
        rw [hn_eq, hn_pow] at *
        -- (2 * (n/2)).testBit (j+1) = (n/2).testBit j; (2 * (n/2) + 1).testBit (j+1) = (n/2).testBit j
        rw [show (2 * (n / 2)) = (n / 2) * 2 from Nat.mul_comm _ _]
        rw [show (n / 2) * 2 = (n / 2) <<< 1 from by rw [Nat.shiftLeft_eq]; ring]
        rw [Nat.testBit_shiftLeft]
        simp
        -- Now: (n / 2).testBit ((j + 1) - 1) = true
        rw [show j + 1 - 1 = j from rfl]
        -- We have (2 * (n / 2) + 1).testBit (j+1) = true (from hi_n)
        have hi_n' : Nat.testBit (2 * (n / 2) + 1) (j + 1) = true := hi_n
        rw [show (2 * (n / 2) + 1) = (n/2) <<< 1 + 1 from by rw [Nat.shiftLeft_eq]; ring] at hi_n'
        rw [show (n / 2) <<< 1 + 1 = (n / 2) <<< 1 ||| 1 from by
            rw [Nat.or_one_eq_add_one_of_testBit_zero_false]
            sorry] at hi_n'
        sorry
      exact absurd (h_testBit i h_sub_bit) (by rw [hi_n]; decide)
    sorry

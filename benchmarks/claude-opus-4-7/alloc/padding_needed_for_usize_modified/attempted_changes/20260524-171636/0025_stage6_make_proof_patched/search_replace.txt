/-- BitVec divisibility bridge: for a power-of-two `a : BitVec 64`, conjunction
    with `a - 1` zero implies modulo `a` is zero. Try focused bv_decide. -/
private theorem bv_div_of_and_pred_zero {x a : BitVec 64}
    (ha : a ≠ 0#64) (hpow : a &&& (a - 1#64) = 0#64)
    (hxand : x &&& (a - 1#64) = 0#64) :
    x % a = 0#64 := by
  revert ha hpow hxand
  bv_decide

/-- Nat-level bit-trick: for any natural `n` and `k`, the conjunction with
    `2 ^ k - 1` extracts the residue modulo `2 ^ k`. Proved by induction
    on `k`, using `Nat.land_succ_succ` (split bit 0) and `Nat.land_zero`. -/
private theorem nat_land_two_pow_sub_one_eq_mod (n k : Nat) :
    n &&& (2 ^ k - 1) = n % 2 ^ k := by
  induction k generalizing n with
  | zero =>
    show n &&& 0 = n % 1
    rw [Nat.land_zero, Nat.mod_one]
  | succ k ih =>
    -- 2 ^ (k+1) - 1 = 2 * (2 ^ k - 1) + 1 = (2 * (2 ^ k) - 1)
    -- Equivalently: 2 ^ (k+1) - 1 = 2 * (2 ^ k) - 1, with low bit 1.
    -- Use the bit-level recurrence on `&&&` via Nat.div2 / Nat.mod2.
    -- Goal: n &&& (2 ^ (k+1) - 1) = n % 2 ^ (k+1).
    have hpos : 0 < 2 ^ k := Nat.two_pow_pos k
    -- Decompose n = 2 * (n / 2) + n % 2.
    -- And 2^(k+1) - 1 = 2 * (2^k - 1) + 1.
    have h_mask_decomp : 2 ^ (k + 1) - 1 = 2 * (2 ^ k - 1) + 1 := by
      rw [Nat.pow_succ]; omega
    have h_n_decomp : n = 2 * (n / 2) + n % 2 := (Nat.div_add_mod n 2).symm
    have h_mod_decomp : n % 2 ^ (k + 1) = 2 * ((n / 2) % 2 ^ k) + n % 2 := by
      rw [Nat.pow_succ, Nat.mul_comm (2 ^ k) 2]
      -- n % (2 * 2^k) = 2 * ((n / 2) % 2^k) + (n % 2)
      conv_lhs => rw [show n = 2 * (n / 2) + n % 2 from h_n_decomp]
      rw [Nat.add_mul_mod_self_left]
      sorry
    sorry

/-- `z &&& m = 0` iff `z` has zero at every bit position where `m` has one. -/
private theorem land_eq_zero_iff_testBit (z m : Nat) :
    z &&& m = 0 ↔ ∀ i, Nat.testBit m i = true → Nat.testBit z i = false := by
  constructor
  · intro h i hm
    have : Nat.testBit (z &&& m) i = false := by rw [h]; rfl
    rw [Nat.testBit_and] at this
    rw [hm] at this
    -- testBit z i && true = false ⇒ testBit z i = false
    simpa using this
  · intro h
    apply Nat.eq_of_testBit_eq
    intro i
    rw [Nat.testBit_and]
    rw [show Nat.testBit 0 i = false from Nat.zero_testBit i]
    -- need: testBit z i && testBit m i = false
    by_cases hm : Nat.testBit m i = true
    · rw [hm, Bool.and_true, h i hm]
    · rw [Bool.eq_false_iff] at hm
      have : Nat.testBit m i = false := by
        cases h : Nat.testBit m i
        · rfl
        · exfalso; exact hm h
      rw [this, Bool.and_false]

    -- need: testBit z i && testBit m i = false

    · rw [Bool.and_false]

private theorem nat_land_two_pow_sub_one_eq_mod (n : Nat) (k : Nat) :

  Nat.and_two_pow_sub_one_eq_mod n k

/-- From the power-of-two characterization, extract a `k` such that
    `align.toNat = 2 ^ k` (and `k ≤ 63`). -/
private theorem pow_two_witness {align : usize}
    (hne : align ≠ 0) (hand : align &&& (align - 1) = 0) :
    ∃ k, k ≤ 63 ∧ align.toNat = 2 ^ k := by
  -- Pass through the BitVec layer: align ≤ 2^63 and align is a power of two,
  -- so the witness is `align.toNat.log2`.
  have hbv : align.toBitVec ≠ 0#64 := fun h => hne (USize64.toBitVec_inj.mp h)
  have hbvand : align.toBitVec &&& (align.toBitVec - 1#64) = 0#64 := by
    show (align &&& (align - 1)).toBitVec = 0#64
    rw [hand]; rfl
  have hLe : align.toNat ≤ 2 ^ 63 := usize_pow_of_two_le hne hand
  have hpos : 0 < align.toNat := by
    rcases Nat.eq_zero_or_pos align.toNat with h | h
    · exfalso; apply hne; apply USize64.toNat_inj.mp; rw [h]; rfl
    · exact h
  -- Nat-level: align.toNat &&& (align.toNat - 1) = 0 (from BitVec).
  have hNatAnd : align.toNat &&& (align.toNat - 1) = 0 := by
    have h1 : (align.toBitVec &&& (align.toBitVec - 1#64)).toNat = 0 := by rw [hbvand]; rfl
    rw [BitVec.toNat_and] at h1
    -- (align.toBitVec - 1#64).toNat = (align.toNat - 1) since align.toNat ≥ 1 < 2^64.
    have h2 : (align.toBitVec - 1#64).toNat = align.toNat - 1 := by
      have := BitVec.toNat_sub align.toBitVec 1#64
      -- toNat_sub: (a - b).toNat = (a.toNat + (2^n - b.toNat)) % 2^n
      -- For b = 1: (2^64 - 1)
      -- For a.toNat ≥ 1: (a.toNat + 2^64 - 1) % 2^64 = a.toNat - 1 since a.toNat < 2^64.
      show ((align.toBitVec - 1#64) : BitVec 64).toNat = align.toNat - 1
      rw [BitVec.toNat_sub]
      have h1tn : (1#64 : BitVec 64).toNat = 1 := rfl
      rw [h1tn]
      have hub : align.toNat < 2 ^ 64 := align.toBitVec.isLt
      show (align.toNat + (2 ^ 64 - 1)) % 2 ^ 64 = align.toNat - 1
      have h_le : align.toNat + 2 ^ 64 - 1 < 2 * 2 ^ 64 := by omega
      have : (align.toNat + (2 ^ 64 - 1)) = align.toNat - 1 + 2 ^ 64 := by omega
      rw [this]
      have : (align.toNat - 1 + 2 ^ 64) % 2 ^ 64 = (align.toNat - 1) % 2 ^ 64 :=
        Nat.add_mod_right _ _
      rw [this]
      have : align.toNat - 1 < 2 ^ 64 := by omega
      exact Nat.mod_eq_of_lt this
    rw [h2] at h1
    exact h1
  -- Standard fact: n > 0 and n &&& (n - 1) = 0 → n = 2 ^ (Nat.log2 n).
  -- We'll inline this. Use Nat.log2 as witness.
  refine ⟨align.toNat.log2, ?_, ?_⟩
  · -- log2 (n) ≤ 63 when n ≤ 2^63.
    have hlog : align.toNat.log2 ≤ 63 := by
      have := Nat.log2_le_log2 hLe (n := 2^63)
      sorry
    exact hlog
  · sorry

    with `a - 1` extracts the BitVec modulo `a`. -/
private theorem bv_and_pred_eq_umod {a : BitVec 64}
    (ha : a ≠ 0#64) (hpow : a &&& (a - 1#64) = 0#64) (x : BitVec 64) :
    x &&& (a - 1#64) = x % a := by
  revert ha hpow
  bv_decide

pub fn padding_needed_for(size: usize, align: usize) -> usize {
    // `Alignment::new(align)` is `None` unless `align` is a power of two.
    if !is_power_of_two_usize(align) {
        // `usize::MAX` extracts to `core_models.num.Impl_11.MAX`, which the
        // Hax Lean prelude does not define. On the 64-bit target this crate
        // targets, `usize::MAX = 2^64 - 1 = 18_446_744_073_709_551_615`; the
        // literal is inlined to avoid the missing identifier (semantics
        // unchanged).
        return 18_446_744_073_709_551_615usize;
    }
    let len_rounded_up = size_rounded_up_to_custom_align(size, align);
    // Cannot overflow because the rounded-up value is never less than `size`.
    len_rounded_up - size
}

// `usize::is_power_of_two` extracts to `core_models.num.Impl_11.is_power_of_two`,
// which the Hax Lean prelude does not model. Inlined with the standard
// bit-trick: `n` is a power of two iff `n != 0 && (n & (n - 1)) == 0`. The
// short-circuit `&&` guards `n - 1` so it never underflows when `n == 0`.
fn is_power_of_two_usize(n: usize) -> bool {
    n != 0 && (n & (n - 1)) == 0
}

fn size_rounded_up_to_custom_align(size: usize, align: usize) -> usize {
    let align_m1 = align - 1;
    (size + align_m1) & !align_m1
}

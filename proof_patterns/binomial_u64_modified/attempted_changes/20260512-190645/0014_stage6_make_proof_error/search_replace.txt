/-- Tight overflow bound for the `n ≤ 67` documented safe range.
    The largest binomial entry under `n ≤ 67` is `natChoose 67 33 ≈
    1.42 × 10^19`, which is strictly below `2^64 ≈ 1.84 × 10^19`. The
    `n ≤ 49` row-sum bound is too weak (since `2^67 > 2^64`); here we
    use `native_decide` over the finite table for `n ≤ 67`, `k ≤ n`. -/
private theorem natChoose_lt_two_pow_64 (n k : Nat) (hn : n ≤ 67) :
    natChoose n k < 2 ^ 64 := by
  by_cases hk : k ≤ n
  · have key : ∀ n' ≤ 67, ∀ k' ≤ n', natChoose n' k' < 2 ^ 64 := by
      native_decide
    exact key n hn k hk
  · push_neg at hk
    rw [natChoose_eq_zero_of_lt n k hk]
    decide

    while d <= k {
        // Measure `k - d`: at iteration entry `d <= k` (loop condition) so
        // `k - d >= 0` (no u64 underflow); each iteration increments `d` so
        // the value strictly decreases by 1. The earlier choice
        // `(k + 1) - d` failed `pureP` synthesis because `grind` cannot
        // disprove `UInt64.addOverflow k 1 = true` without a bound on `k`.
        hax_lib::loop_decreases!(k - d);
        r = multiply_and_divide(r, n, d);
        n = n - 1;
        d = d + 1;
    }

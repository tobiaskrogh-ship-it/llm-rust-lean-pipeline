        // `k - d >= 0` (no u64 underflow); each iteration increments `d` so

        // disprove `UInt64.addOverflow k 1 = true` without a bound on `k`.

        d = d + 1;
    }

    let mut r: u64 = 1;
    let mut d: u64 = 1;
    // The original is `loop { if d > k { break; } ... }`. Hax's bare-`loop`
    // extraction produces a degenerate `let ⟨...⟩ := sorry` (it can't see
    // the `break` as a structured exit), so we rewrite the loop as a `while`
    // with an explicit termination measure. The measure must be a *total*
    // expression on the loop state — `pureTermination` in the Hax `while_loop`
    // combinator is required to type-check under precondition `True`, not
    // under the loop condition. That rules out `k - d` (underflows when
    // `d > k`) and `(k + 1) - d` (the `k + 1` itself can overflow). We
    // therefore introduce an auxiliary counter `iters_left` that starts at
    // `k`, is decremented unconditionally each iteration, and serves as the
    // measure: a single u64 variable that grind can see is total and
    // strictly decreases by 1 each iteration. `d` keeps its original
    // increasing role inside the body so the call to
    // `multiply_and_divide(r, n, d)` is unchanged from the source.
    let mut iters_left: u64 = k;
    while iters_left > 0 {
        hax_lib::loop_decreases!(iters_left);
        r = multiply_and_divide(r, n, d);
        n = n - 1;
        d = d + 1;
        iters_left = iters_left - 1;
    }
    r
}

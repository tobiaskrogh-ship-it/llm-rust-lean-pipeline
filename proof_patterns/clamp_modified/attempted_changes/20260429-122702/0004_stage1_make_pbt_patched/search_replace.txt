    prop_compose! {
        /// Yields (x, lo, hi) with lo ≤ hi and x < lo (lower-clamp zone).
        /// Generated with zero rejects: lo is drawn first, hi ≥ lo second,
        /// and x < lo third.  `lo` is re-bound via Just so it is available
        /// in the function body (proptest does not forward outer vars automatically).
        fn below_bound()
            (lo in 1u8..=255u8)
            (hi in lo..=255u8, x in 0u8..lo, lo in Just(lo))
            -> (u8, u8, u8)
        {
            (x, lo, hi)
        }
    }

    prop_compose! {

theorem sqrt_lower_bound (a : u64) :
    ⦃ ⌜ True ⌝ ⦄
    nth_root_u64.sqrt_u64 a
    ⦃ ⇓ r => ⌜ r.toNat ^ 2 ≤ a.toNat ⌝ ⦄ := by
  -- Stage 2 conversion to equational form (per `while_example/README.md`).
  rw [RustM.Triple_iff_BitVec]
  simp only [decide_true, Bool.not_true, Bool.false_or, Bool.and_eq_true,
             decide_eq_true_eq]
  -- Case split on the cascade of guards inside `sqrt_u64`.
  by_cases ha0 : a = 0
  · -- a = 0 branch: returns pure 0; `0^2 = 0 ≤ 0` ✓
    subst ha0
    refine ⟨?_, ?_⟩ <;> decide
  · by_cases ha4 : a.toNat < 4
    · -- 1 ≤ a < 4 branch: returns pure 1; `1^2 = 1 ≤ a` since `1 ≤ a.toNat`.
      have ha_pos : 0 < a.toNat := by
        rcases Nat.eq_zero_or_pos a.toNat with h | h
        · exfalso; apply ha0; apply UInt64.toNat_inj.mp; rw [h]; rfl
        · exact h
      -- Show `sqrt_u64 a` reduces to `pure 1`.
      have h_a_lt : a <ᵤ (4 : u64) = true := by
        rw [UInt64.lt_iff_toNat_lt]
        show a.toNat < (4 : u64).toNat
        rw [show (4 : u64).toNat = 4 from rfl]
        exact ha4
      -- Stuck sub-goal: showing `(sqrt_u64 a).toBVRustM = ⟨true, 1⟩` for

      -- does not fire without a target-specific `pure_toBVRustM` lemma.

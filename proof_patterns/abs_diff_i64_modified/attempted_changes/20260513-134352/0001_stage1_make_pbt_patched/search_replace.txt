pub fn abs_diff(a: i64, b: i64) -> i64 {
    if a > b {
        a - b
    } else {
        b - a
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use proptest::prelude::*;

    proptest! {
        /// Postcondition, `a > b` branch.
        ///
        /// When `a > b` and the difference `a - b` fits in `i64` (the
        /// precondition that makes the inner subtraction overflow-free), the
        /// function returns exactly `a - b`.
        #[test]
        fn abs_diff_returns_a_minus_b_when_a_gt_b(a: i64, b: i64) {
            // Branch guard.
            prop_assume!(a > b);
            // Precondition: the subtraction `a - b` must not overflow `i64`.
            // Evaluated in `i128` to avoid the very overflow we are guarding
            // against.
            prop_assume!((a as i128) - (b as i128) <= i64::MAX as i128);

            prop_assert_eq!(abs_diff(a, b), a - b);
        }

        /// Postcondition, `a <= b` branch.
        ///
        /// When `a <= b` and the difference `b - a` fits in `i64`, the
        /// function returns exactly `b - a`. Covers the equality case
        /// `a == b` (where the result is `0`) as well.
        #[test]
        fn abs_diff_returns_b_minus_a_when_a_le_b(a: i64, b: i64) {
            prop_assume!(a <= b);
            prop_assume!((b as i128) - (a as i128) <= i64::MAX as i128);

            prop_assert_eq!(abs_diff(a, b), b - a);
        }
    }

    /// Failure condition.
    ///
    /// When `|a - b|` exceeds `i64::MAX`, the chosen branch's subtraction
    /// overflows and the function panics (under the default debug-overflow
    /// checks that `cargo test` uses). `abs_diff(i64::MAX, i64::MIN)` takes
    /// the `a > b` branch and computes `i64::MAX - i64::MIN`, which
    /// overflows.
    #[test]
    #[should_panic]
    fn abs_diff_panics_when_difference_exceeds_i64_max() {
        let _ = abs_diff(i64::MAX, i64::MIN);
    }
}

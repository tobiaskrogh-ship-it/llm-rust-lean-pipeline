theorem gcd_while_zero_b (b : u64) :
    gcd_while.gcd_while 0 b = RustM.ok b := by
  by_cases hb : b = 0
  · subst hb; exact gcd_while_a_zero 0
  · unfold gcd_while.gcd_while
    unfold rust_primitives.hax.while_loop
    unfold Lean.Loop.MonoLoopCombinator.while_loop
    -- First iteration: condition is true (b ≠ 0), so we run the body.
    unfold Lean.Loop.MonoLoopCombinator.forIn
    unfold Lean.Loop.MonoLoopCombinator.forIn.loop
    unfold Lean.Loop.loopCombinator
    simp only [bind, ExceptT.bind, Option.bind, pure, ExceptT.pure,
               rust_primitives.cmp.ne, rust_primitives.ops.arith.Rem.rem,
               instRemRustPrimitivesOpsArithUInt64]
    -- After the simp, we should have a term involving the next loop iteration
    -- with state (b, 0).
    have hb' : (b == 0) = false := by simp [hb]
    simp [hb', hb]
    -- Second iteration: condition is false (the new b = 0 = 0), so loop exits.
    unfold Lean.Loop.MonoLoopCombinator.forIn.loop
    unfold Lean.Loop.loopCombinator
    simp [bind, ExceptT.bind, Option.bind, pure, ExceptT.pure,
          rust_primitives.cmp.ne]
    rfl

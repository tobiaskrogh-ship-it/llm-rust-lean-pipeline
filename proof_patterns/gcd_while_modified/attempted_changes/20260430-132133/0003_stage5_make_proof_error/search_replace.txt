theorem gcd_while_total (a b : u64) :
    ∃ r : u64, gcd_while.gcd_while a b = RustM.ok r := by
  -- Stronger Hoare-triple lemma: the function returns `RustM.ok r` for some `r`.
  suffices h : ⦃ ⌜ True ⌝ ⦄ gcd_while.gcd_while a b ⦃ ⇓ _ => ⌜ True ⌝ ⦄ by
    -- Extract a value from the Hoare triple.
    sorry
  hax_mvcgen [gcd_while.gcd_while]
  · exact ⟨fun _ => True, by intro b; mvcgen⟩
  · exact ⟨fun ⟨_, b⟩ => b.toNat, by intro b; mvcgen⟩
  · exact ⟨fun ⟨_, b⟩ => decide (b ≠ 0), by intro ⟨a, b⟩ _; mvcgen⟩
  · sorry  -- step
  · sorry  -- final

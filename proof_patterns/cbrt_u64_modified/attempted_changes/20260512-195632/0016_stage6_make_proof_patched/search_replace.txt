private theorem cbrt_postcondition_u32_branch (x : u64)
    (hu32 : x ≤ (4294967295 : u64)) (hnsmall : ¬ x < (8 : u64)) :
    ∃ r : u64, cbrt_u64.cbrt x = RustM.ok r ∧
               r.toNat ^ 3 ≤ x.toNat ∧
               x.toNat < (r.toNat + 1) ^ 3 := by
  -- Substantive attempt: reduce the function in this branch by selecting
  -- the right `if`-branches and unfolding the surrounding casts, exposing
  -- the inner call to `cbrt_u32` on `x.toUInt32`. The remaining work is
  -- then totality + cubic-bound correctness of `cbrt_u32`, which is the
  -- actual stuck sub-goal.
  have h_x_le : x.toNat ≤ 4294967295 := UInt64.le_iff_toNat_le.mp hu32
  -- Step 1: unfold the function and fire the two outer `if`s using the
  -- `decide` rewrites for the comparisons.
  simp only [cbrt_u64.cbrt, rust_primitives.cmp.lt, rust_primitives.cmp.le,
             pure_bind, decide_eq_false hnsmall, decide_eq_true hu32,
             if_false, if_true, rust_primitives.hax.cast_op]
  -- After Step 1, the goal exposes the chain
  --     `do let a ← Cast.cast x; let y ← cbrt_u32 a; Cast.cast y`
  -- where `Cast.cast : UInt64 → RustM UInt32 = pure x.toUInt32` and
  -- `Cast.cast : UInt32 → RustM UInt64 = pure y.toUInt64`. Both
  -- collapse the surrounding binds via `pure_bind`.
  --
  -- Stuck sub-goal: this leaves
  --     `∃ r, (cbrt_u32 x.toUInt32 >>= fun y => pure y.toUInt64) = RustM.ok r ∧
  --           r.toNat^3 ≤ x.toNat ∧ x.toNat < (r.toNat+1)^3`,
  -- which requires `cbrt_u32`'s combined spec
  --     `∃ y, cbrt_u32 a = RustM.ok y ∧ y.toNat^3 ≤ a.toNat ∧ a.toNat < (y.toNat+1)^3`.
  -- The Stage-1 Hoare triple over the 11-iteration `Tuple4`-state loop
  -- requires the Hacker's-Delight icbrt2 cubic invariant — not in the
  -- prelude (see header docstring for the full enumeration). No tactic
  -- short of supplying that lemma will close the remaining goal.
  sorry

private theorem cbrt_guess_u64_correct (a : u64) (ha : 4294967295 < a.toNat) :
    ∃ g : u64, cbrt_u64.cbrt_guess_u64 a = RustM.ok g ∧
               g.toNat ≤ 4294967296 ∧
               a.toNat ≤ g.toNat ^ 3 := by
  -- Substantive attempt with real tactics. Strategy: convert to a
  -- Hoare triple, then prove the triple via the two-stage `while_example`
  -- pattern over the two sequential `while_loop`s.
  suffices h : ⦃⌜4294967295 < a.toNat⌝⦄
        cbrt_u64.cbrt_guess_u64 a
      ⦃⇓ g => ⌜g.toNat ≤ 4294967296 ∧ a.toNat ≤ g.toNat ^ 3⌝⦄ by
    rw [RustM.Triple_iff_BitVec] at h
    rw [show decide (4294967295 < a.toNat) = true from decide_eq_true ha] at h
    simp only [Bool.not_true, Bool.false_or,
               Bool.and_eq_true, decide_eq_true_eq] at h
    obtain ⟨hok, hval⟩ := h
    cases hf : cbrt_u64.cbrt_guess_u64 a with
    | none =>
      rw [hf] at hok; simp [RustM.toBVRustM] at hok
    | some result =>
      cases result with
      | ok v =>
        rw [hf] at hval
        simp [RustM.toBVRustM] at hval
        exact ⟨v, congrArg RustM.ok rfl, hval.1, hval.2⟩
      | error e =>
        rw [hf] at hok
        cases e <;> simp [RustM.toBVRustM] at hok
  -- The remaining Hoare-triple goal is the actual hard work.
  -- Stuck at: discharging Loop 1's body-step VC. The required invariant
  -- is
  --   `y.toNat * 2^hi.toNat ≤ a.toNat ∧
  --    a.toNat < (y.toNat + 1) * 2^hi.toNat ∧
  --    hi.toNat ≤ 64`,
  -- which exactly captures `hi = floor(log2 a)` at loop exit (when y=1).
  -- The exit invariant must then feed Loop 2's invariant
  --   `g.toNat = 2^i.toNat ∧ i.toNat ≤ k.toNat ∧ k.toNat = (hi.toNat + 3)/3`,
  -- and finally combine to give `a.toNat ≤ g.toNat^3`.
  --
  -- The cube property `a.toNat ≤ g.toNat^3` reduces algebraically to
  -- `a < 2^(3k) = 2^(3*⌈(hi+1)/3⌉) ≥ 2^(hi+1)`, which holds because
  -- `a < 2^(hi+1)` (from the log2 invariant).
  --
  -- Structural unblock: `Nat.log2_via_shift_loop` lemma packaging the
  -- shift-loop-counts-bits identity. Without it, the Stage-1 triple
  -- for Loop 1 cannot establish `hi = floor(log2 a)`, and downstream
  -- the cube-bound cannot follow.
  sorry

/// Hacker's-Delight `icbrt2`, monomorphized to `u32`.
///
/// Mirrors the body of the `Roots::cbrt` impl for `u32` produced by
/// `unsigned_roots!(u32)` in `src/roots.rs:333..351`.
fn cbrt_u32(a: u32) -> u32 {
    let mut x = a;
    let mut y2: u32 = 0;
    let mut y: u32 = 0;
    let smax: u32 = 32 / 3; // bits::<u32>() / 3 = 10
    let mut s_iter = smax + 1;
    while s_iter > 0 {
        s_iter -= 1;
        let s = s_iter * 3;
        y2 *= 4;
        y *= 2;
        let b = 3 * (y2 + y) + 1;
        if (x >> s) >= b {
            x -= b << s;
            y2 += 2 * y + 1;
            y += 1;
        }
    }
    y
}

/// Defunctionalized `fixpoint(guess, |x| (a/(x*x) + x*2) / 3)` from
/// `src/roots.rs:373..374` (the `next` closure for cube roots).
fn fixpoint_cbrt(a: u64, mut x: u64) -> u64 {
    let mut xn = (a / (x * x) + x * 2) / 3;
    while x < xn {
        x = xn;
        xn = (a / (x * x) + x * 2) / 3;
    }
    while x > xn {
        x = xn;
        xn = (a / (x * x) + x * 2) / 3;
    }
    x
}

/// Concrete `u64` cube root --- truncated principal `∛x`.
///
/// `cbrt(x)` returns the largest `r: u64` with `r*r*r <= x`.
pub fn cbrt(x: u64) -> u64 {
    let a = x;

    // Source has `if a < 8 { return (a > 0) as $T; }` --- inlined.
    if a < 8 {
        return if a > 0 { 1 } else { 0 };
    }

    // Source has `if a <= core::u32::MAX as $T { return (a as u32).cbrt() as $T; }`.
    // We inline the `<u32 as Roots>::cbrt` body via `cbrt_u32`.
    if a <= u32::MAX as u64 {
        return cbrt_u32(a as u32) as u64;

/// Hacker's-Delight `icbrt2`, monomorphized to `u32`.
///
/// Mirrors the body of the `Roots::cbrt` impl for `u32` produced by
/// `unsigned_roots!(u32)` in `src/roots.rs:333..351`.
fn cbrt_u32(a: u32) -> u32 {
    let mut x = a;
    let mut y2: u32 = 0;
    let mut y: u32 = 0;
    let smax: u32 = 32 / 3; // bits::<u32>() / 3 = 10
    let mut s_iter = smax + 1;
    while s_iter > 0 {
        hax_lib::loop_decreases!(s_iter);
        s_iter -= 1;
        let s = s_iter * 3;
        y2 *= 4;
        y *= 2;
        let b = 3 * (y2 + y) + 1;
        if (x >> s) >= b {
            x -= b << s;
            y2 += 2 * y + 1;
            y += 1;
        }
    }
    y
}

/// Defunctionalized `fixpoint(guess, |x| (a/(x*x) + x*2) / 3)` from
/// `src/roots.rs:373..374` (the `next` closure for cube roots).
///
/// Termination: integer Newton iteration on `f(x) = (a/x^2 + 2x)/3`
/// monotonically converges to `floor(cbrt(a))`. The first loop walks
/// upward while `xn > x`, the second walks downward while `xn < x`,
/// and termination is reached when `xn == x` (a fixed point) or when
/// the iteration stops decreasing (a 2-cycle near the answer, broken
/// by the `>` comparison). Both loops decrease `|x - xn|` until they
/// stop, but to give Hax a strictly-monotone integer measure we use
/// `xn` itself for the first loop (it is increasing toward `x` from
/// below, so `MAX - xn` decreases) and `x` itself for the second
/// loop (it is decreasing). For `loop_decreases!` we need a
/// strictly-decreasing measure: in the first loop, `xn > x` is the
/// guard, and after `x = xn; xn = next(x)`, the new `xn` is at most
/// the old `xn` (Newton is a contraction once near the answer); we
/// pick `u64::MAX - xn` which strictly decreases as long as `xn`
/// strictly increases. In the second loop, `x` strictly decreases.
fn fixpoint_cbrt(a: u64, mut x: u64) -> u64 {
    let mut xn = (a / (x * x) + x * 2) / 3;
    while x < xn {
        hax_lib::loop_decreases!(u64::MAX - xn);
        x = xn;
        xn = (a / (x * x) + x * 2) / 3;
    }
    while x > xn {
        hax_lib::loop_decreases!(x);
        x = xn;
        xn = (a / (x * x) + x * 2) / 3;
    }
    x
}

/// Compute the bit length of `a` (number of bits required to represent
/// `a`, with `bit_length(0) = 0`). Replaces `u64::leading_zeros` /
/// `u64::ilog2`, neither of which Hax extracts cleanly.
fn bit_length_u64(a: u64) -> u32 {
    let mut bits: u32 = 0;
    let mut tmp: u64 = a;
    while tmp > 0 {
        hax_lib::loop_decreases!(tmp);
        tmp >>= 1;
        bits += 1;
    }
    bits
}

/// Concrete `u64` cube root --- truncated principal `∛x`.
///
/// `cbrt(x)` returns the largest `r: u64` with `r*r*r <= x`.
pub fn cbrt(x: u64) -> u64 {
    let a = x;

    // Source has `if a < 8 { return (a > 0) as $T; }` --- inlined.
    if a < 8 {
        return if a > 0 { 1 } else { 0 };
    }

    // Source has `if a <= core::u32::MAX as $T { return (a as u32).cbrt() as $T; }`.
    // We inline the `<u32 as Roots>::cbrt` body via `cbrt_u32`.
    // `u32::MAX` is inlined as the literal `4_294_967_295u64` because
    // `u32::MAX` does not extract through Hax (becomes an undefined
    // `core_models.num.Impl_8.MAX` symbol in the Lean output).
    if a <= 4_294_967_295u64 {
        return cbrt_u32(a as u32) as u64;
    }

    // Integer-only initial guess for the Newton fixpoint, replacing the
    // `#[cfg(feature = "std")]` `(a as f64).cbrt() as u64` branch of the
    // source. Hax does not support `f64` casts or `f64::cbrt`, so we use
    // the `#[cfg(not(feature = "std"))]` style guess from `num-integer`:
    // shift `1` left by `(bit_length(a) + 2) / 3`, which is always
    // `>= cbrt(a)` and `<= 1 << 22`, so `guess * guess <= 1 << 44` and
    // the first iteration of `fixpoint_cbrt` does not overflow.
    let bits = bit_length_u64(a);
    let guess: u64 = 1u64 << ((bits + 2) / 3);
    fixpoint_cbrt(a, guess)
}

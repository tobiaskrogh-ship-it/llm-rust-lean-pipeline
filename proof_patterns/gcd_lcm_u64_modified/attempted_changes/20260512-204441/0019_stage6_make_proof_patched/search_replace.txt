/-- Two and any odd number are coprime. Used in `nat_gcd_two_mul_odd`. -/
private theorem nat_coprime_two_odd (b : Nat) :
    Nat.Coprime 2 (2 * b + 1) := by
  show Nat.gcd 2 (2 * b + 1) = 1
  rw [Nat.gcd_rec]
  have h_mod : (2 * b + 1) % 2 = 1 := by
    rw [Nat.add_mul_mod_self_left]
  rw [h_mod]
  decide

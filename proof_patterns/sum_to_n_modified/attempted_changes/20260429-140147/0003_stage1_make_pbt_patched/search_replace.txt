#[cfg(test)]
mod tests {
    use super::*;

    /// Base case of the recursion: the empty sum is 0.
    /// Captures the explicit `if n == 0 { 0 }` branch — a postcondition
    /// for the single input n = 0.
    #[test]
    fn base_case_zero() {
        assert_eq!(sum_to_n(0), 0);
    }

    /// Main postcondition: for any valid `n`, the result equals the
    /// Gauss closed-form `n*(n+1)/2`. This is the central semantic
    /// claim about the function. We sweep a contiguous range of `n`
    /// rather than only sampling, so any off-by-one or wrong-branch
    /// bug is caught.
    ///
    /// `n` is bounded well below the stack-overflow / arithmetic-overflow
    /// region so the precondition holds for every sampled value.
    #[test]
    fn matches_closed_form() {
        for n in 0u64..=2_000 {
            let expected = n * (n + 1) / 2;
            assert_eq!(sum_to_n(n), expected, "failed at n = {n}");
        }
    }

    /// A handful of hand-computed values. These are redundant in
    /// principle with `matches_closed_form`, but they guard against
    /// the case where the closed-form expression itself is mis-stated
    /// in the test (a bug-in-the-oracle failure mode that the property
    /// test alone would not catch).
    #[test]
    fn known_values() {
        assert_eq!(sum_to_n(1), 1);
        assert_eq!(sum_to_n(2), 3);
        assert_eq!(sum_to_n(10), 55);
        assert_eq!(sum_to_n(100), 5050);
    }
}
